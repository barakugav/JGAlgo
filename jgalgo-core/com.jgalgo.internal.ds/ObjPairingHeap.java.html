package com.jgalgo.internal.ds;

import java.util.Iterator;
import com.jgalgo.internal.util.Assertions;
import com.jgalgo.internal.util.JGAlgoUtils;
import java.util.Comparator;


/**
 * A Pairing heap implementation.
 *
 * <p>
 * A pointer based heap implementation that support almost any operation in \(O(1)\) amortized time, except
 * {@link #remove(ObjReferenceableHeap.Ref)} and {@link #decreaseKey(ObjReferenceableHeap.Ref, Object)}
 * which takes \(O(\log n)\) time amortized.
 *
 * <p>
 * Using this heap, {@link ShortestPathSingleSourceDijkstra} can be implemented in time \(O(m + n \log n)\) rather than
 * \(O(m \log n)\) as the {@link #decreaseKey(ObjReferenceableHeap.Ref, Object)} operation is performed in
 * \(O(1)\) time amortized.
 *
 * <p>
 * Pairing heaps are one of the best pointer based heaps implementations in practice, and should be used as a default
 * choice for the general use case.
 *
 * @param  <K> the keys type
 * @see    <a href="https://en.wikipedia.org/wiki/Pairing_heap">Wikipedia</a>
 * @author Barak Ugav
 */
public class ObjPairingHeap <K> extends PairingHeapBase<ObjPairingHeap.Node <K>> implements ObjReferenceableHeap <K> {

	private final Comparator <K> c;

	public ObjPairingHeap(Comparator <K> c) {
		this.c = c;
	}

	@Override
	public Comparator <K> comparator() {
		return c;
	}

	@Override
	public ObjReferenceableHeap.Ref <K> findMin() {
		Assertions.notEmpty(this);
		return minRoot;
	}

	void insertNode(Node <K> n) {
		if (minRoot == null) {
			minRoot = n;
		} else {
			minRoot = meld(minRoot, n);
		}
	}

	void afterKeyDecrease(Node <K> n) {
		if (n == minRoot)
			return;
		cut(n);
		minRoot = meld(minRoot, n);
	}

	@Override
	public ObjReferenceableHeap.Ref <K> extractMin() {
		Node <K> min = minRoot;
		removeRoot();
		return min;
	}

	@Override
	public void remove(ObjReferenceableHeap.Ref <K> ref) {
		Node <K> n = (Node <K>) ref;
		assert minRoot != null;
		if (n != minRoot) {
			cut(n);
			addChild(n, minRoot);
			minRoot = n;
		}
		removeRoot();
	}

	@Override
	public void meld(ObjReferenceableHeap <K> heap) {
		Assertions.heapNoMeldWithSelf(this, heap);
		Assertions.heapMeldWithSameImpl(getClass(), heap);
		Assertions.heapEqualComparatorBeforeMeld(comparator(), heap.comparator());
		ObjPairingHeap <K> h = (ObjPairingHeap <K>) heap;

		if (minRoot == null) {
			minRoot = h.minRoot;
		} else if (h.minRoot != null) {
			minRoot = meld(minRoot, h.minRoot);
		}

		h.minRoot = null;
	}

	private Node <K> meld(Node <K> n1, Node <K> n2) {
		return c == null ? meldDefaultCmp(n1, n2) : meldCustomCmp(n1, n2);
	}

	Node <K> meldDefaultCmp(Node <K> n1, Node <K> n2) {
		assert n1.prevOrParent == null && n1.next == null;
		assert n2.prevOrParent == null && n2.next == null;

		/* assume n1 has smaller key than n2 */
		if (JGAlgoUtils.cmpDefault(n1.key, n2.key) > 0) {
			Node <K> temp = n1;
			n1 = n2;
			n2 = temp;
		}

		addChild(n1, n2);
		return n1;
	}

	Node <K> meldCustomCmp(Node <K> n1, Node <K> n2) {
		assert n1.prevOrParent == null && n1.next == null;
		assert n2.prevOrParent == null && n2.next == null;

		/* assume n1 has smaller key than n2 */
		if (c.compare(n1.key, n2.key) > 0) {
			Node <K> temp = n1;
			n1 = n2;
			n2 = temp;
		}

		addChild(n1, n2);
		return n1;
	}

	@SuppressWarnings({ "unchecked", "rawtypes" })
	@Override
	public Iterator<ObjReferenceableHeap.Ref <K>> iterator() {
		return (Iterator) new PairingHeapBase.PreOrderIter<>(minRoot);
	}

	@Override
	public ObjReferenceableHeap.Ref <K> insert(K key) {
		Node <K> n = new Node <>(key);
		insertNode(n);
		return n;
	}


	private void removeRoot() {
		if (minRoot.child == null) {
			minRoot = null;
			return;
		}

		if (c == null) {
			/* meld pairs from left to right */
			Node <K> tail;
			for (Node <K> prev = minRoot, next = minRoot.child;;) {
				Node <K> n1 = next;
				if (n1 == null) {
					tail = prev;
					break;
				}

				Node <K> n2 = n1.next;
				if (n2 == null) {
					n1.prevOrParent = prev;
					tail = n1;
					break;
				}
				next = n2.next;
				n1.next = null;
				n2.next = null;
				n1.prevOrParent = null;
				n2.prevOrParent = null;
				n1 = meldDefaultCmp(n1, n2);

				n1.prevOrParent = prev;
				prev = n1;
			}
			minRoot.child = null;

			/* meld all from right to left */
			Node <K> root = tail, prev = root.prevOrParent;
			root.prevOrParent = null;
			for (;;) {
				Node <K> other = prev;
				if (other == minRoot) {
					minRoot = root;
					break;
				}
				prev = other.prevOrParent;
				other.prevOrParent = null;
				root = meldDefaultCmp(root, other);
			}
		} else {

			/* meld pairs from left to right */
			Node <K> tail;
			for (Node <K> prev = minRoot, next = minRoot.child;;) {
				Node <K> n1 = next;
				if (n1 == null) {
					tail = prev;
					break;
				}

				Node <K> n2 = n1.next;
				if (n2 == null) {
					n1.prevOrParent = prev;
					tail = n1;
					break;
				}
				next = n2.next;
				n1.next = null;
				n2.next = null;
				n1.prevOrParent = null;
				n2.prevOrParent = null;
				n1 = meldCustomCmp(n1, n2);

				n1.prevOrParent = prev;
				prev = n1;
			}
			minRoot.child = null;

			/* meld all from right to left */
			Node <K> root = tail, prev = root.prevOrParent;
			root.prevOrParent = null;
			for (;;) {
				Node <K> other = prev;
				if (other == minRoot) {
					minRoot = root;
					break;
				}
				prev = other.prevOrParent;
				other.prevOrParent = null;
				root = meldCustomCmp(root, other);
			}
		}
	}

	@Override
	public void decreaseKey(ObjReferenceableHeap.Ref <K> ref, K newKey) {
		Node <K> n = (Node <K>) ref;
		Assertions.heapDecreaseKeyIsSmaller(n.key, newKey, c);
		n.key = newKey;
		afterKeyDecrease(n);
	}

	@Override
	public void increaseKey(ObjReferenceableHeap.Ref <K> ref, K newKey) {
		Node <K> n = (Node <K>) ref;
		Assertions.heapIncreaseKeyIsGreater(n.key, newKey, c);
		remove(n);
		n.key = newKey;
		insertNode(n);
	}

	static class Node <K> extends PairingHeapBase.NodeBase<Node <K>> implements ObjReferenceableHeap.Ref <K> {
		K key;

		Node(K key) {
			this.key = key;
		}

		@Override
		public K key() {
			return key;
		}


		@Override
		void clearKeyData() {
			key = null;
		}

		@Override
		public String toString() {
			return String.valueOf(key);
		}
	}

}
