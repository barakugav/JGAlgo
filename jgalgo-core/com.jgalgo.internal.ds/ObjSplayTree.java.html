package com.jgalgo.internal.ds;

import java.util.Comparator;
import com.jgalgo.internal.util.JGAlgoUtils;

import java.util.Iterator;
import com.jgalgo.internal.util.Assertions;

/**
 * Splay binary search tree.
 *
 * <p>
 * The splay tree is a binary search tree which is not strictly balanced, in contrast to a red black tree. The splay
 * tree has a property that recent accessed elements are closer to the root, and therefore faster to access. All
 * operations have a time complexity of \(O(\log n)\) amortized time.
 *
 * <p>
 * The splay tree is specifically efficient for splits and joins operations.
 *
 * <p>
 * Based on 'Self-Adjusting Binary Search Trees' by Sleator and Tarjan (1985).
 *
 * @param  <K> the keys type
 * @see        <a href="https://en.wikipedia.org/wiki/Splay_tree">Wikipedia</a>
 * @author     Barak Ugav
 */
public class ObjSplayTree <K> implements ObjBinarySearchTree <K> {

	private final Comparator <K> c;
	private SplayBSTNode <K> root;
	private final SplayImpl<K,  SplayBSTNode <K>> impl = new SplayImpl<>();

	/**
	 * Constructs a new, empty splay tree, ordered according to the natural ordering of its keys.
	 *
	 * <p>
	 * All keys inserted into the tree must implement the {@link Comparable} interface. Furthermore, all such keys must
	 * be <i>mutually comparable</i>: {@code k1.compareTo(k2)} must not throw a {@code ClassCastException} for any keys
	 * {@code k1} and {@code k2} in the tree. If the user attempts to insert a key to the tree that violates this
	 * constraint (for example, the user attempts to insert a string element to a tree whose keys are integers), the
	 * {@code insert} call will throw a {@code ClassCastException}.
	 */
	public ObjSplayTree() {
		this(null);
	}

	/**
	 * Constructs a new, empty splay tree, with keys ordered according to the specified comparator.
	 *
	 * <p>
	 * All keys inserted into the tree must be <i>mutually comparable</i> by the specified comparator:
	 * {@code comparator.compare(k1, k2)} must not throw a {@code ClassCastException} for any keys {@code k1} and
	 * {@code k2} in the tree. If the user attempts to insert a key to the tree that violates this constraint, the
	 * {@code insert} call will throw a {@code ClassCastException}.
	 *
	 * @param comparator the comparator that will be used to order this tree. If {@code null}, the
	 *                       {@linkplain Comparable natural ordering} of the keys will be used.
	 */
	public ObjSplayTree(Comparator <K> comparator) {
		this.c = comparator;
	}

	@Override
	public boolean isEmpty() {
		return root == null;
	}

	@Override
	public boolean isNotEmpty() {
		return root != null;
	}

	@Override
	public Comparator <K> comparator() {
		return c;
	}

	@Override
	public ObjReferenceableHeap.Ref <K> extractMin() {
		ObjReferenceableHeap.Ref <K> min = findMin();
		remove(min);
		return min;
	}

	@Override
	public ObjReferenceableHeap.Ref <K> extractMax() {
		ObjReferenceableHeap.Ref <K> max = findMax();
		remove(max);
		return max;
	}

	@SuppressWarnings({ "unchecked", "rawtypes" })
	@Override
	public Iterator<ObjReferenceableHeap.Ref <K>> iterator() {
		return (Iterator) new ObjBinarySearchTrees.BSTIterator <>(root);
	}

	@Override
	public ObjReferenceableHeap.Ref <K> insert(K key) {
		return insertNode(new SplayBSTNode <>(key));
	}

	private SplayBSTNode <K> insertNode(SplayBSTNode <K> n) {
		assert n.parent == null;
		assert n.left == null;
		assert n.right == null;
		if (root == null)
			return root = n;

		ObjBinarySearchTrees.insert(root, c, n);
		return root = impl.splay(n);
	}

	@Override
	public void remove(ObjReferenceableHeap.Ref <K> ref) {
		SplayBSTNode <K> n = (SplayBSTNode <K>) ref;

		/* If the node has two children, swap with successor */
		if (n.hasLeftChild() && n.hasRightChild())
			ObjBinarySearchTrees.swap(n, ObjBinarySearchTrees.getSuccessor(n));

		SplayBSTNode <K> child;
		if (!n.hasLeftChild()) {
			replace(n, child = n.right);
		} else {
			assert !n.hasRightChild();
			replace(n, child = n.left);
		}


		SplayBSTNode <K> parent = n.parent;
		n.clearWithoutUserData();
		root = impl.splay(child != null ? child : parent);
	}

	private static <K> void replace(SplayBSTNode <K> u, SplayBSTNode <K> v) {
		/* replaces u with v */
		if (u.parent != null) {
			if (u.isLeftChild()) {
				u.parent.left = v;
			} else {
				assert u.isRightChild();
				u.parent.right = v;
			}
		}
		if (v != null)
			v.parent = u.parent;
	}

	@Override
	public void decreaseKey(ObjReferenceableHeap.Ref <K> ref, K newKey) {
		SplayBSTNode <K> n = (SplayBSTNode <K>) ref;
		Assertions.Heaps.decreaseKeyIsSmaller(n.key, newKey, c);
		remove(n);
		n.key = newKey;
		insertNode(n);
	}

	@Override
	public ObjReferenceableHeap.Ref <K> find(K key) {
		SplayBSTNode <K> n = ObjBinarySearchTrees.find(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public ObjReferenceableHeap.Ref <K> findMin() {
		checkTreeNotEmpty();
		return root = impl.splay(ObjBinarySearchTrees.findMin(root));
	}

	@Override
	public ObjReferenceableHeap.Ref <K> findMax() {
		checkTreeNotEmpty();
		return root = impl.splay(ObjBinarySearchTrees.findMax(root));
	}

	@Override
	public ObjReferenceableHeap.Ref <K> findOrSmaller(K key) {
		SplayBSTNode <K> n = ObjBinarySearchTrees.findOrSmaller(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public ObjReferenceableHeap.Ref <K> findOrGreater(K key) {
		SplayBSTNode <K> n = ObjBinarySearchTrees.findOrGreater(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public ObjReferenceableHeap.Ref <K> findSmaller(K key) {
		SplayBSTNode <K> n = ObjBinarySearchTrees.findSmaller(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public ObjReferenceableHeap.Ref <K> findGreater(K key) {
		SplayBSTNode <K> n = ObjBinarySearchTrees.findGreater(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public ObjReferenceableHeap.Ref <K> getPredecessor(ObjReferenceableHeap.Ref <K> ref) {
		SplayBSTNode <K> n = ObjBinarySearchTrees.getPredecessor((SplayBSTNode <K>) ref);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public ObjReferenceableHeap.Ref <K> getSuccessor(ObjReferenceableHeap.Ref <K> ref) {
		SplayBSTNode <K> n = ObjBinarySearchTrees.getSuccessor((SplayBSTNode <K>) ref);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public void meld(ObjReferenceableHeap <K> heap) {
		Assertions.Heaps.noMeldWithSelf(this, heap);
		Assertions.Heaps.meldWithSameImpl(ObjSplayTree.class, heap);
		Assertions.Heaps.equalComparatorBeforeMeld(comparator(), heap.comparator());
		ObjSplayTree <K> h = (ObjSplayTree <K>) heap;
		if (h.isEmpty())
			return;
		if (isEmpty()) {
			root = h.root;
			h.root = null;
			return;
		}

		Comparator <K> cmp = c != null ? c : (a, b) -> JGAlgoUtils.cmpDefault(a, b);

		K min1, max1, min2, max2;
		if (cmp.compare(max1 = findMax().key(), min2 = h.findMin().key()) <= 0) {
			/* all elements in this tree are <= than all elements in other tree */
			root = meld(this, h);
		} else if (cmp.compare(min1 = findMin().key(), max2 = h.findMax().key()) >= 0) {
			/* all elements in this tree are >= than all elements in other tree */
			root = meld(h, this);
		} else {
			int minCmp = cmp.compare(min1, min2);
			int maxCmp = cmp.compare(max1, max2);
			ObjSplayTree <K> hLow = null, hHigh = null;

			if (minCmp < 0) {
				hLow = this.splitSmaller(min2);
			} else if (minCmp > 0) {
				hLow = h.splitSmaller(min1);
			}
			if (maxCmp < 0) {
				hHigh = h.splitGreater(max1);
			} else if (maxCmp > 0) {
				hHigh = this.splitGreater(max2);
			}

			if (h.root != null) {
				/* there is nothing smarter to do than 'addAll' for the shared range */
				/* We use 'insertNode' instead of 'insert' to maintain user references to nodes */
				for (SplayBSTNode <K> node = h.root;;) {
					for (;;) {
						while (node.hasLeftChild())
							node = node.left;
						if (!node.hasRightChild())
							break;
						node = node.right;
					}
					SplayBSTNode <K> parent = node.parent;
					if (parent == null) {
						insertNode(node);
						break;
					} else {
						if (parent.right == node) {
							parent.right = null;
						} else {
							assert parent.left == node;
							parent.left = null;
						}
						node.parent = null;
						insertNode(node);
						node = parent;
					}
				}
			}

			if (hLow != null) {
				assert cmp.compare(hLow.findMax().key(), findMin().key()) < 0;
				root = meld(hLow, this);
			}
			if (hHigh != null) {
				assert cmp.compare(hHigh.findMin().key(), findMax().key()) > 0;
				root = meld(this, hHigh);
			}
		}
		h.root = null;
	}

	private static <K> SplayBSTNode <K> meld(ObjSplayTree <K> t1, ObjSplayTree <K> t2) {
		/* Assume all nodes in t1 are smaller than all nodes in t2 */

		/* Splay t1 max and t2 min */
		SplayBSTNode <K> n1 = (SplayBSTNode <K>) t1.findMax();
		assert n1.isRoot();
		assert !n1.hasRightChild();

		n1.right = t2.root;
		t2.root.parent = n1;
		return n1;
	}

	@Override
	public ObjSplayTree <K> splitSmaller(K key) {
		ObjSplayTree <K> newTree = new ObjSplayTree <>(c);
		SplayBSTNode <K> pred = (SplayBSTNode <K>) findSmaller(key);
		if (pred == null)
			return newTree;
		assert pred.isRoot();

		if ((root = pred.right) != null) {
			pred.right.parent = null;
			pred.right = null;
		}

		newTree.root = pred;
		return newTree;
	}

	@Override
	public ObjSplayTree <K> splitGreater(K key) {
		SplayBSTNode <K> succ = (SplayBSTNode <K>) findGreater(key);
		if (succ == null)
			return new ObjSplayTree <>(c);
		return split(succ);
	}

	@Override
	public ObjSplayTree <K> split(ObjReferenceableHeap.Ref <K> ref) {
		ObjSplayTree <K> newTree = new ObjSplayTree <>(c);

		SplayBSTNode <K> n = impl.splay((SplayBSTNode <K>) ref);
		assert n.isRoot();

		if ((root = n.left) != null) {
			n.left.parent = null;
			n.left = null;
		}

		newTree.root = n;
		return newTree;
	}

	@Override
	public void clear() {
		if (root == null)
			return;
		ObjBinarySearchTrees.clear(root);
		root = null;
	}

	static class BaseNode<K,  N extends BaseNode<K,  N>> extends ObjBinarySearchTrees.Node<K,  N> {

		BaseNode(K key) {
			super(key);
		}

		public K key() {
			return key;
		}

	}

	static class SplayImpl<K,  N extends BaseNode<K,  N>> {

		SplayImpl() {}

		N splay(N n) {
			if (n == null || n.isRoot())
				return n;
			for (;;) {
				N parent = n.parent;

				if (parent.isRoot()) {
					/* zig */
					rotate(n);
					return n;
				}

				if (n.isLeftChild() == parent.isLeftChild()) {
					/* zig-zig */
					rotate(parent);
					rotate(n);
				} else {
					/* zig-zag */
					rotate(n);
					rotate(n);
				}

				if (n.isRoot())
					return n;
			}
		}

		private void rotate(N n) {
			N parent = n.parent, grandparent = parent.parent;

			beforeRotate(n);

			if (n.isLeftChild()) {

				parent.left = n.right;
				if (parent.hasLeftChild())
					parent.left.parent = parent;
				n.right = parent;

			} else {
				assert n.isRightChild();

				parent.right = n.left;
				if (parent.hasRightChild())
					parent.right.parent = parent;
				n.left = parent;
			}

			n.parent = grandparent;
			parent.parent = n;
			if (grandparent != null) {
				if (grandparent.left == parent)
					grandparent.left = n;
				else
					grandparent.right = n;
			}
		}

		void beforeRotate(N n) {}

	}

	private static class SplayBSTNode <K> extends BaseNode<K,  SplayBSTNode <K>> implements ObjReferenceableHeap.Ref <K> {


		SplayBSTNode(K key) {
			super(key);
		}

		@Override
		public K key() {
			return key;
		}


		@Override
		public String toString() {
			return String.valueOf(key);
		}

	}

	private void checkTreeNotEmpty() {
		if (root == null)
			throw new IllegalStateException("Tree is empty");
	}

}
