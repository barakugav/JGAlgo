package com.jgalgo.internal.ds;

import java.util.Arrays;
import com.jgalgo.internal.util.JGAlgoUtils;
import java.util.Comparator;
import java.util.ConcurrentModificationException;
import java.util.Iterator;
import com.jgalgo.internal.util.Assertions;

/**
 * A binomial heap implementation.
 *
 * <p>
 * Pointer based data structure that support user references to the internal nodes, allowing efficient \(O(\log n)\)
 * implementation of the {@link #remove} and {@link #decreaseKey} operations. The regular operations like
 * {@link #insert}, {@link #extractMin()} and {@link #findMin()} are also implemented in \(O(\log n)\) time. Another
 * advantage of the binomial heap is its ability to merge with another binomial heap in \(O(\log n)\) time, which is
 * much faster than the required \(O(n)\) time of binary heaps.
 *
 * <p>
 * Although it has great complexities bounds, {@link #decreaseKey} can is implemented in faster times by Fibonnachi and
 * Pairing heaps.
 *
 * @param  <K> the keys type
 * @see        <a href="https://en.wikipedia.org/wiki/Binomial_heap">Wikipedia</a>
 * @author     Barak Ugav
 */
public class ObjBinomialHeap <K> implements ObjReferenceableHeap <K> {

	private final Comparator <K> c;
	private Node <K>[] roots;
	private int rootsLen;

	@SuppressWarnings("unchecked")
	private Node <K>[] tempNodeArray = EmptyNodeArr;

	@SuppressWarnings("rawtypes")
	private static final Node[] EmptyNodeArr = new Node[0];

	/**
	 * Constructs a new, empty binomial heap, ordered according to the natural ordering of its keys.
	 *
	 * <p>
	 * All keys inserted into the heap must implement the {@link Comparable} interface. Furthermore, all such keys must
	 * be <i>mutually comparable</i>: {@code k1.compareTo(k2)} must not throw a {@code ClassCastException} for any keys
	 * {@code k1} and {@code k2} in the heap. If the user attempts to insert a key to the heap that violates this
	 * constraint (for example, the user attempts to insert a string element to a heap whose keys are integers), the
	 * {@code insert} call will throw a {@code ClassCastException}.
	 */
	public ObjBinomialHeap() {
		this(null);
	}

	/**
	 * Constructs a new, empty binomial heap, with keys ordered according to the specified comparator.
	 *
	 * <p>
	 * All keys inserted into the heap must be <i>mutually comparable</i> by the specified comparator:
	 * {@code comparator.compare(k1, k2)} must not throw a {@code ClassCastException} for any keys {@code k1} and
	 * {@code k2} in the heap. If the user attempts to insert a key to the heap that violates this constraint, the
	 * {@code insert} call will throw a {@code ClassCastException}.
	 *
	 * @param comparator the comparator that will be used to order this heap. If {@code null}, the
	 *                       {@linkplain Comparable natural ordering} of the keys will be used.
	 */
	@SuppressWarnings("unchecked")
	public ObjBinomialHeap(Comparator <K> comparator) {
		c = comparator;
		roots = EmptyNodeArr;
		rootsLen = 0;
	}

	@Override
	public Comparator <K> comparator() {
		return c;
	}

	@Override
	public ObjReferenceableHeap.Ref <K> extractMin() {
		ObjReferenceableHeap.Ref <K> min = findMin();
		remove(min);
		return min;
	}

	private void swapParentChild(Node <K> parent, Node <K> child) {
		final Node <K> pNext = parent.next, pPrev = parent.prev, pParent = parent.parent, pChild = parent.child;
		Node <K> t;

		parent.next = (t = child.next);
		if (t != null)
			t.prev = parent;
		parent.prev = (t = child.prev);
		if (t != null)
			t.next = parent;
		parent.child = child.child;
		for (Node <K> p = child.child; p != null; p = p.next)
			p.parent = parent;

		child.next = pNext;
		if (pNext != null)
			pNext.prev = child;
		child.prev = pPrev;
		if (pPrev != null)
			pPrev.next = child;
		child.child = pChild == child ? parent : pChild;
		for (Node <K> p = child.child; p != null; p = p.next)
			p.parent = child;
		child.parent = pParent;
		if (pParent != null && pParent.child == parent)
			pParent.child = child;

		if (pParent == null) {
			/* Switched a root, fix roots array */
			Node <K>[] rs = roots;
			for (int i = 0; i < rootsLen; i++) {
				if (rs[i] == parent) {
					rs[i] = child;
					break;
				}
			}
		}
	}

	@Override
	public void decreaseKey(ObjReferenceableHeap.Ref <K> ref, K newKey) {
		Node <K> node = (Node <K>) ref;
		Assertions.Heaps.decreaseKeyIsSmaller(node.key, newKey, c);
		node.key = newKey;

		if (c == null) {
			for (Node <K> p; (p = node.parent) != null;) {
				if (JGAlgoUtils.cmpDefault(p.key, newKey) <= 0)
					break;
				swapParentChild(p, node);
			}
		} else {
			for (Node <K> p; (p = node.parent) != null;) {
				if (c.compare(p.key, newKey) <= 0)
					break;
				swapParentChild(p, node);
			}
		}
	}

	@Override
	public void remove(ObjReferenceableHeap.Ref <K> ref) {
		Node <K> node = (Node <K>) ref;

		/* propagate to top of the tree */
		for (Node <K> p; (p = node.parent) != null;)
			swapParentChild(p, node);

		Node <K>[] rs = roots;
		int rootIdx = -1;
		for (int i = 0; i < rootsLen; i++) {
			if (rs[i] == node) {
				rs[i] = null;
				rootIdx = i;
				break;
			}
		}
		if (rootIdx == -1)
			throw new ConcurrentModificationException();

		Node <K>[] childs = tempArray(rootIdx);
		Node <K> next, p = node.child;
		for (int i = 0; i < rootIdx; i++, p = next) {
			next = p.next;
			p.parent = null;
			p.next = null;
			p.prev = null;
			childs[rootIdx - i - 1] = p;
		}

		meld(childs, rootIdx);
		while (rootsLen > 0 && roots[rootsLen - 1] == null)
			rootsLen--;
	}

	@Override
	public void clear() {
		Node <K>[] rs = roots;
		int rslen = rootsLen;

		for (int i = 0; i < rslen; i++) {
			if (rs[i] != null) {
				Trees.clear(rs[i], n -> {
					n.key = null;
				});
				rs[i] = null;
			}
		}

		rootsLen = 0;
	}

	@Override
	public boolean isEmpty() {
		return rootsLen == 0;
	}

	@Override
	public boolean isNotEmpty() {
		return rootsLen != 0;
	}

	@Override
	public ObjReferenceableHeap.Ref <K> insert(K key) {
		Node <K> node = new Node <>(key);
		Node <K>[] h2 = tempArray(1);
		h2[0] = node;
		meld(h2, 1);
		return node;
	}

	@SuppressWarnings({ "unchecked", "rawtypes" })
	@Override
	public Iterator<ObjReferenceableHeap.Ref <K>> iterator() {
		return (Iterator) new Itr();
	}

	private static <K> Node <K> mergeTreesDefaultCmp(Node <K> r1, Node <K> r2) {
		assert r1 != r2;
		if (r1 == r2)
			throw new IllegalStateException();
		if (JGAlgoUtils.cmpDefault(r1.key, r2.key) > 0) {
			Node <K> t = r1;
			r1 = r2;
			r2 = t;
		}
		r2.next = r1.child;
		Node <K> next = r1.child;
		if (next != null)
			next.prev = r2;
		r1.child = r2;
		r2.parent = r1;

		return r1;
	}

	private Node <K> mergeTreesCustomCmp(Node <K> r1, Node <K> r2) {
		assert r1 != r2;
		if (r1 == r2)
			throw new IllegalStateException();
		if (c.compare(r1.key, r2.key) > 0) {
			Node <K> t = r1;
			r1 = r2;
			r2 = t;
		}
		r2.next = r1.child;
		Node <K> next = r1.child;
		if (next != null)
			next.prev = r2;
		r1.child = r2;
		r2.parent = r1;

		return r1;
	}

	private int meld(Node <K>[] rs2, int rs2len) {
		Node <K>[] rs1 = roots;
		int rs1len = rootsLen;
		int h2size = 0;
		rootsLen = rs1len > rs2len ? rs1len : rs2len;

		int maxArrLen = Math.max(rs1len, rs2len) + 1;
		if (maxArrLen > roots.length)
			roots = Arrays.copyOf(roots, Math.max(maxArrLen, roots.length * 2));

		Node <K> carry = null;
		if (c == null) {
			for (int i = 0; i < rootsLen; i++) {
				Node <K> r1 = i < rs1len ? rs1[i] : null;
				Node <K> r2 = i < rs2len ? rs2[i] : null;

				if (r2 != null)
					h2size += 1 << i;

				if ((r1 == null && r2 == null) || (r1 != null && r2 != null)) {
					roots[i] = carry;
					carry = (r1 != null && r2 != null) ? mergeTreesDefaultCmp(r1, r2) : null;
				} else {
					Node <K> r = r1 != null ? r1 : r2;
					if (carry == null)
						roots[i] = r;
					else {
						roots[i] = null;
						carry = mergeTreesDefaultCmp(carry, r);
					}
				}
			}
		} else {
			for (int i = 0; i < rootsLen; i++) {
				Node <K> r1 = i < rs1len ? rs1[i] : null;
				Node <K> r2 = i < rs2len ? rs2[i] : null;

				if (r2 != null)
					h2size += 1 << i;

				if ((r1 == null && r2 == null) || (r1 != null && r2 != null)) {
					roots[i] = carry;
					carry = (r1 != null && r2 != null) ? mergeTreesCustomCmp(r1, r2) : null;
				} else {
					Node <K> r = r1 != null ? r1 : r2;
					if (carry == null)
						roots[i] = r;
					else {
						roots[i] = null;
						carry = mergeTreesCustomCmp(carry, r);
					}
				}
			}
		}
		if (carry != null)
			roots[rootsLen++] = carry;

		Arrays.fill(rs2, 0, rs2len, null);

		return h2size;
	}

	@SuppressWarnings("unchecked")
	@Override
	public void meld(ObjReferenceableHeap <K> heap) {
		Assertions.Heaps.noMeldWithSelf(this, heap);
		Assertions.Heaps.meldWithSameImpl(ObjBinomialHeap.class, heap);
		Assertions.Heaps.equalComparatorBeforeMeld(comparator(), heap.comparator());

		ObjBinomialHeap <K> h = (ObjBinomialHeap <K>) heap;
		meld(h.roots, h.rootsLen);

		h.roots = EmptyNodeArr;
		h.rootsLen = 0;
	}

	@Override
	public ObjReferenceableHeap.Ref <K> findMin() {
		Assertions.Heaps.notEmpty(this);
		Node <K>[] rs = roots;
		int rsLen = rootsLen;
		Node <K> min = null;

		if (c == null) {
			for (int i = 0; i < rsLen; i++)
				if (rs[i] != null && (min == null || JGAlgoUtils.cmpDefault(min.key, rs[i].key) > 0))
					min = rs[i];
		} else {
			for (int i = 0; i < rsLen; i++)
				if (rs[i] != null && (min == null || c.compare(min.key, rs[i].key) > 0))
					min = rs[i];
		}
		return min;
	}

	private Node <K>[] tempArray(int n) {
		if (n > tempNodeArray.length)
			tempNodeArray = Arrays.copyOf(tempNodeArray, Math.max(2, Math.max(tempNodeArray.length, n)));
		return tempNodeArray;
	}

	private static class Node <K> extends Trees.TreeNodeImpl<Node <K>> implements ObjReferenceableHeap.Ref <K> {

		K key;

		Node(K key) {
			this.key = key;
		}

		@Override
		public K key() {
			return key;
		}


		@Override
		public String toString() {
			return String.valueOf(key);
		}

	}

	private class Itr extends Trees.PreOrderIter<Node <K>> {

		private int nextRootIdx;

		Itr() {
			super(null);
			nextRootIdx = 0;
			advanceToNextRoot();
		}

		private boolean advanceToNextRoot() {
			for (;; nextRootIdx++) {
				if (nextRootIdx >= rootsLen)
					return false;
				if (roots[nextRootIdx] != null) {
					reset(roots[nextRootIdx]);
					nextRootIdx++;
					return true;
				}
			}
		}

		@Override
		boolean advance() {
			return super.advance() || advanceToNextRoot();
		}

	}

}
