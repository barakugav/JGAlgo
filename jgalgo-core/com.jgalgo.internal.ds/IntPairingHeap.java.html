package com.jgalgo.internal.ds;

import java.util.Iterator;
import com.jgalgo.internal.util.Assertions;
import it.unimi.dsi.fastutil.ints.IntComparator;


/**
 * A Pairing heap implementation.
 *
 * <p>
 * A pointer based heap implementation that support almost any operation in \(O(1)\) amortized time, except
 * {@link #remove(IntReferenceableHeap.Ref)} and {@link #decreaseKey(IntReferenceableHeap.Ref, int)}
 * which takes \(O(\log n)\) time amortized.
 *
 * <p>
 * Using this heap, {@link ShortestPathSingleSourceDijkstra} can be implemented in time \(O(m + n \log n)\) rather than
 * \(O(m \log n)\) as the {@link #decreaseKey(IntReferenceableHeap.Ref, int)} operation is performed in
 * \(O(1)\) time amortized.
 *
 * <p>
 * Pairing heaps are one of the best pointer based heaps implementations in practice, and should be used as a default
 * choice for the general use case.
 *
 * @see    <a href="https://en.wikipedia.org/wiki/Pairing_heap">Wikipedia</a>
 * @author Barak Ugav
 */
public class IntPairingHeap  extends PairingHeapBase<IntPairingHeap.Node > implements IntReferenceableHeap  {

	private final IntComparator  c;

	public IntPairingHeap(IntComparator  c) {
		this.c = c;
	}

	@Override
	public IntComparator  comparator() {
		return c;
	}

	@Override
	public IntReferenceableHeap.Ref  findMin() {
		Assertions.notEmpty(this);
		return minRoot;
	}

	void insertNode(Node  n) {
		if (minRoot == null) {
			minRoot = n;
		} else {
			minRoot = meld(minRoot, n);
		}
	}

	void afterKeyDecrease(Node  n) {
		if (n == minRoot)
			return;
		cut(n);
		minRoot = meld(minRoot, n);
	}

	@Override
	public IntReferenceableHeap.Ref  extractMin() {
		Node  min = minRoot;
		removeRoot();
		return min;
	}

	@Override
	public void remove(IntReferenceableHeap.Ref  ref) {
		Node  n = (Node ) ref;
		assert minRoot != null;
		if (n != minRoot) {
			cut(n);
			addChild(n, minRoot);
			minRoot = n;
		}
		removeRoot();
	}

	@Override
	public void meld(IntReferenceableHeap  heap) {
		Assertions.heapNoMeldWithSelf(this, heap);
		Assertions.heapMeldWithSameImpl(getClass(), heap);
		Assertions.heapEqualComparatorBeforeMeld(comparator(), heap.comparator());
		IntPairingHeap  h = (IntPairingHeap ) heap;

		if (minRoot == null) {
			minRoot = h.minRoot;
		} else if (h.minRoot != null) {
			minRoot = meld(minRoot, h.minRoot);
		}

		h.minRoot = null;
	}

	private Node  meld(Node  n1, Node  n2) {
		return c == null ? meldDefaultCmp(n1, n2) : meldCustomCmp(n1, n2);
	}

	Node  meldDefaultCmp(Node  n1, Node  n2) {
		assert n1.prevOrParent == null && n1.next == null;
		assert n2.prevOrParent == null && n2.next == null;

		/* assume n1 has smaller key than n2 */
		if (n1.key > n2.key) {
			Node  temp = n1;
			n1 = n2;
			n2 = temp;
		}

		addChild(n1, n2);
		return n1;
	}

	Node  meldCustomCmp(Node  n1, Node  n2) {
		assert n1.prevOrParent == null && n1.next == null;
		assert n2.prevOrParent == null && n2.next == null;

		/* assume n1 has smaller key than n2 */
		if (c.compare(n1.key, n2.key) > 0) {
			Node  temp = n1;
			n1 = n2;
			n2 = temp;
		}

		addChild(n1, n2);
		return n1;
	}

	@SuppressWarnings({ "unchecked", "rawtypes" })
	@Override
	public Iterator<IntReferenceableHeap.Ref > iterator() {
		return (Iterator) new PairingHeapBase.PreOrderIter<>(minRoot);
	}

	@Override
	public IntReferenceableHeap.Ref  insert(int key) {
		Node  n = new Node (key);
		insertNode(n);
		return n;
	}


	private void removeRoot() {
		if (minRoot.child == null) {
			minRoot = null;
			return;
		}

		if (c == null) {
			/* meld pairs from left to right */
			Node  tail;
			for (Node  prev = minRoot, next = minRoot.child;;) {
				Node  n1 = next;
				if (n1 == null) {
					tail = prev;
					break;
				}

				Node  n2 = n1.next;
				if (n2 == null) {
					n1.prevOrParent = prev;
					tail = n1;
					break;
				}
				next = n2.next;
				n1.next = null;
				n2.next = null;
				n1.prevOrParent = null;
				n2.prevOrParent = null;
				n1 = meldDefaultCmp(n1, n2);

				n1.prevOrParent = prev;
				prev = n1;
			}
			minRoot.child = null;

			/* meld all from right to left */
			Node  root = tail, prev = root.prevOrParent;
			root.prevOrParent = null;
			for (;;) {
				Node  other = prev;
				if (other == minRoot) {
					minRoot = root;
					break;
				}
				prev = other.prevOrParent;
				other.prevOrParent = null;
				root = meldDefaultCmp(root, other);
			}
		} else {

			/* meld pairs from left to right */
			Node  tail;
			for (Node  prev = minRoot, next = minRoot.child;;) {
				Node  n1 = next;
				if (n1 == null) {
					tail = prev;
					break;
				}

				Node  n2 = n1.next;
				if (n2 == null) {
					n1.prevOrParent = prev;
					tail = n1;
					break;
				}
				next = n2.next;
				n1.next = null;
				n2.next = null;
				n1.prevOrParent = null;
				n2.prevOrParent = null;
				n1 = meldCustomCmp(n1, n2);

				n1.prevOrParent = prev;
				prev = n1;
			}
			minRoot.child = null;

			/* meld all from right to left */
			Node  root = tail, prev = root.prevOrParent;
			root.prevOrParent = null;
			for (;;) {
				Node  other = prev;
				if (other == minRoot) {
					minRoot = root;
					break;
				}
				prev = other.prevOrParent;
				other.prevOrParent = null;
				root = meldCustomCmp(root, other);
			}
		}
	}

	@Override
	public void decreaseKey(IntReferenceableHeap.Ref  ref, int newKey) {
		Node  n = (Node ) ref;
		Assertions.heapDecreaseKeyIsSmaller(n.key, newKey, c);
		n.key = newKey;
		afterKeyDecrease(n);
	}

	@Override
	public void increaseKey(IntReferenceableHeap.Ref  ref, int newKey) {
		Node  n = (Node ) ref;
		Assertions.heapIncreaseKeyIsGreater(n.key, newKey, c);
		remove(n);
		n.key = newKey;
		insertNode(n);
	}

	static class Node  extends PairingHeapBase.NodeBase<Node > implements IntReferenceableHeap.Ref  {
		int key;

		Node(int key) {
			this.key = key;
		}

		@Override
		public int key() {
			return key;
		}



		@Override
		public String toString() {
			return String.valueOf(key);
		}
	}

}
