package com.jgalgo.internal.ds;

import java.util.Iterator;
import com.jgalgo.internal.util.Assertions;
import it.unimi.dsi.fastutil.doubles.DoubleComparator;


/**
 * A Pairing heap implementation.
 *
 * <p>
 * A pointer based heap implementation that support almost any operation in \(O(1)\) amortized time, except
 * {@link #remove(DoubleObjReferenceableHeap.Ref)} and {@link #decreaseKey(DoubleObjReferenceableHeap.Ref, double)}
 * which takes \(O(\log n)\) time amortized.
 *
 * <p>
 * Using this heap, {@link ShortestPathSingleSourceDijkstra} can be implemented in time \(O(m + n \log n)\) rather than
 * \(O(m \log n)\) as the {@link #decreaseKey(DoubleObjReferenceableHeap.Ref, double)} operation is performed in
 * \(O(1)\) time amortized.
 *
 * <p>
 * Pairing heaps are one of the best pointer based heaps implementations in practice, and should be used as a default
 * choice for the general use case.
 *
 * @param  <V> the values type
 * @see    <a href="https://en.wikipedia.org/wiki/Pairing_heap">Wikipedia</a>
 * @author Barak Ugav
 */
public class DoubleObjPairingHeap <V> extends PairingHeapBase<DoubleObjPairingHeap.Node <V>> implements DoubleObjReferenceableHeap <V> {

	private final DoubleComparator  c;

	public DoubleObjPairingHeap(DoubleComparator  c) {
		this.c = c;
	}

	@Override
	public DoubleComparator  comparator() {
		return c;
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> findMin() {
		Assertions.notEmpty(this);
		return minRoot;
	}

	void insertNode(Node <V> n) {
		if (minRoot == null) {
			minRoot = n;
		} else {
			minRoot = meld(minRoot, n);
		}
	}

	void afterKeyDecrease(Node <V> n) {
		if (n == minRoot)
			return;
		cut(n);
		minRoot = meld(minRoot, n);
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> extractMin() {
		Node <V> min = minRoot;
		removeRoot();
		return min;
	}

	@Override
	public void remove(DoubleObjReferenceableHeap.Ref <V> ref) {
		Node <V> n = (Node <V>) ref;
		assert minRoot != null;
		if (n != minRoot) {
			cut(n);
			addChild(n, minRoot);
			minRoot = n;
		}
		removeRoot();
	}

	@Override
	public void meld(DoubleObjReferenceableHeap <V> heap) {
		Assertions.heapNoMeldWithSelf(this, heap);
		Assertions.heapMeldWithSameImpl(getClass(), heap);
		Assertions.heapEqualComparatorBeforeMeld(comparator(), heap.comparator());
		DoubleObjPairingHeap <V> h = (DoubleObjPairingHeap <V>) heap;

		if (minRoot == null) {
			minRoot = h.minRoot;
		} else if (h.minRoot != null) {
			minRoot = meld(minRoot, h.minRoot);
		}

		h.minRoot = null;
	}

	private Node <V> meld(Node <V> n1, Node <V> n2) {
		return c == null ? meldDefaultCmp(n1, n2) : meldCustomCmp(n1, n2);
	}

	Node <V> meldDefaultCmp(Node <V> n1, Node <V> n2) {
		assert n1.prevOrParent == null && n1.next == null;
		assert n2.prevOrParent == null && n2.next == null;

		/* assume n1 has smaller key than n2 */
		if (n1.key > n2.key) {
			Node <V> temp = n1;
			n1 = n2;
			n2 = temp;
		}

		addChild(n1, n2);
		return n1;
	}

	Node <V> meldCustomCmp(Node <V> n1, Node <V> n2) {
		assert n1.prevOrParent == null && n1.next == null;
		assert n2.prevOrParent == null && n2.next == null;

		/* assume n1 has smaller key than n2 */
		if (c.compare(n1.key, n2.key) > 0) {
			Node <V> temp = n1;
			n1 = n2;
			n2 = temp;
		}

		addChild(n1, n2);
		return n1;
	}

	@SuppressWarnings({ "unchecked", "rawtypes" })
	@Override
	public Iterator<DoubleObjReferenceableHeap.Ref <V>> iterator() {
		return (Iterator) new PairingHeapBase.PreOrderIter<>(minRoot);
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> insert(double key) {
		Node <V> n = new Node <>(key);
		insertNode(n);
		return n;
	}


	private void removeRoot() {
		if (minRoot.child == null) {
			minRoot = null;
			return;
		}

		if (c == null) {
			/* meld pairs from left to right */
			Node <V> tail;
			for (Node <V> prev = minRoot, next = minRoot.child;;) {
				Node <V> n1 = next;
				if (n1 == null) {
					tail = prev;
					break;
				}

				Node <V> n2 = n1.next;
				if (n2 == null) {
					n1.prevOrParent = prev;
					tail = n1;
					break;
				}
				next = n2.next;
				n1.next = null;
				n2.next = null;
				n1.prevOrParent = null;
				n2.prevOrParent = null;
				n1 = meldDefaultCmp(n1, n2);

				n1.prevOrParent = prev;
				prev = n1;
			}
			minRoot.child = null;

			/* meld all from right to left */
			Node <V> root = tail, prev = root.prevOrParent;
			root.prevOrParent = null;
			for (;;) {
				Node <V> other = prev;
				if (other == minRoot) {
					minRoot = root;
					break;
				}
				prev = other.prevOrParent;
				other.prevOrParent = null;
				root = meldDefaultCmp(root, other);
			}
		} else {

			/* meld pairs from left to right */
			Node <V> tail;
			for (Node <V> prev = minRoot, next = minRoot.child;;) {
				Node <V> n1 = next;
				if (n1 == null) {
					tail = prev;
					break;
				}

				Node <V> n2 = n1.next;
				if (n2 == null) {
					n1.prevOrParent = prev;
					tail = n1;
					break;
				}
				next = n2.next;
				n1.next = null;
				n2.next = null;
				n1.prevOrParent = null;
				n2.prevOrParent = null;
				n1 = meldCustomCmp(n1, n2);

				n1.prevOrParent = prev;
				prev = n1;
			}
			minRoot.child = null;

			/* meld all from right to left */
			Node <V> root = tail, prev = root.prevOrParent;
			root.prevOrParent = null;
			for (;;) {
				Node <V> other = prev;
				if (other == minRoot) {
					minRoot = root;
					break;
				}
				prev = other.prevOrParent;
				other.prevOrParent = null;
				root = meldCustomCmp(root, other);
			}
		}
	}

	@Override
	public void decreaseKey(DoubleObjReferenceableHeap.Ref <V> ref, double newKey) {
		Node <V> n = (Node <V>) ref;
		Assertions.heapDecreaseKeyIsSmaller(n.key, newKey, c);
		n.key = newKey;
		afterKeyDecrease(n);
	}

	@Override
	public void increaseKey(DoubleObjReferenceableHeap.Ref <V> ref, double newKey) {
		Node <V> n = (Node <V>) ref;
		Assertions.heapIncreaseKeyIsGreater(n.key, newKey, c);
		remove(n);
		n.key = newKey;
		insertNode(n);
	}

	static class Node <V> extends PairingHeapBase.NodeBase<Node <V>> implements DoubleObjReferenceableHeap.Ref <V> {
		double key;
		private V value;

		Node(double key) {
			this.key = key;
		}

		@Override
		public double key() {
			return key;
		}

		@Override
		public V value() {
			return value;
		}

		@Override
		public void setValue(V val) {
			value = val;
		}

		@Override
		void clearKeyData() {
			value = null;
		}

		@Override
		public String toString() {
			return "{" + key + ":" + value + "}";
		}
	}

}
