package com.jgalgo.internal.ds;

import it.unimi.dsi.fastutil.doubles.DoubleComparator;

import java.util.Iterator;
import com.jgalgo.internal.util.Assertions;

/**
 * Splay binary search tree.
 *
 * <p>
 * The splay tree is a binary search tree which is not strictly balanced, in contrast to a red black tree. The splay
 * tree has a property that recent accessed elements are closer to the root, and therefore faster to access. All
 * operations have a time complexity of \(O(\log n)\) amortized time.
 *
 * <p>
 * The splay tree is specifically efficient for splits and joins operations.
 *
 * <p>
 * Based on 'Self-Adjusting Binary Search Trees' by Sleator and Tarjan (1985).
 *
 * @param  <V> the values type
 * @see        <a href="https://en.wikipedia.org/wiki/Splay_tree">Wikipedia</a>
 * @author     Barak Ugav
 */
public class DoubleObjSplayTree <V> implements DoubleObjBinarySearchTree <V> {

	private final DoubleComparator  c;
	private SplayBSTNode <V> root;
	private final SplayImplWithSize <V> impl = new SplayImplWithSize <>();

	/**
	 * Constructs a new, empty splay tree, ordered according to the natural ordering of its keys.
	 */
	public DoubleObjSplayTree() {
		this(null);
	}

	/**
	 * Constructs a new, empty splay tree, with keys ordered according to the specified comparator.
	 *
	 * @param comparator the comparator that will be used to order this tree. If {@code null}, the
	 *                       {@linkplain Comparable natural ordering} of the keys will be used.
	 */
	public DoubleObjSplayTree(DoubleComparator  comparator) {
		this.c = comparator;
	}

	@Override
	public boolean isEmpty() {
		return root == null;
	}

	@Override
	public boolean isNotEmpty() {
		return root != null;
	}

	@Override
	public DoubleComparator  comparator() {
		return c;
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> extractMin() {
		DoubleObjReferenceableHeap.Ref <V> min = findMin();
		remove(min);
		return min;
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> extractMax() {
		DoubleObjReferenceableHeap.Ref <V> max = findMax();
		remove(max);
		return max;
	}

	@SuppressWarnings({ "unchecked", "rawtypes" })
	@Override
	public Iterator<DoubleObjReferenceableHeap.Ref <V>> iterator() {
		return (Iterator) new DoubleBinarySearchTrees.BSTIterator <>(root);
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> insert(double key) {
		return insertNode(new SplayBSTNode <>(key));
	}

	private SplayBSTNode <V> insertNode(SplayBSTNode <V> n) {
		assert n.parent == null;
		assert n.left == null;
		assert n.right == null;
		if (root == null)
			return root = n;

		DoubleBinarySearchTrees.insert(root, c, n);
		for (SplayBSTNode <V> p = n.parent; p != null; p = p.parent)
			p.size++;
		return root = impl.splay(n);
	}

	@Override
	public void remove(DoubleObjReferenceableHeap.Ref <V> ref) {
		SplayBSTNode <V> n = (SplayBSTNode <V>) ref;

		/* If the node has two children, swap with successor */
		if (n.hasLeftChild() && n.hasRightChild())
			swap(n, DoubleBinarySearchTrees.getSuccessor(n));

		SplayBSTNode <V> child;
		if (!n.hasLeftChild()) {
			replace(n, child = n.right);
		} else {
			assert !n.hasRightChild();
			replace(n, child = n.left);
		}

		/* Decrease ancestors size by 1 */
		for (SplayBSTNode <V> p = n.parent; p != null; p = p.parent)
			p.size--;

		SplayBSTNode <V> parent = n.parent;
		n.clearWithoutUserData();
		root = impl.splay(child != null ? child : parent);
	}

	private static <V> void replace(SplayBSTNode <V> u, SplayBSTNode <V> v) {
		/* replaces u with v */
		if (u.parent != null) {
			if (u.isLeftChild()) {
				u.parent.left = v;
			} else {
				assert u.isRightChild();
				u.parent.right = v;
			}
		}
		if (v != null)
			v.parent = u.parent;
	}

	private void swap(SplayBSTNode <V> n1, SplayBSTNode <V> n2) {
		DoubleBinarySearchTrees.swap(n1, n2);
		if (n1 == root)
			root = n2;
		else if (n2 == root)
			root = n1;
		int size = n1.size;
		n1.size = n2.size;
		n2.size = size;
	}

	@Override
	public void decreaseKey(DoubleObjReferenceableHeap.Ref <V> ref, double newKey) {
		SplayBSTNode <V> n = (SplayBSTNode <V>) ref;
		Assertions.Heaps.decreaseKeyIsSmaller(n.key, newKey, c);
		remove(n);
		n.key = newKey;
		insertNode(n);
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> find(double key) {
		SplayBSTNode <V> n = DoubleBinarySearchTrees.find(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> findMin() {
		checkTreeNotEmpty();
		return root = impl.splay(DoubleBinarySearchTrees.findMin(root));
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> findMax() {
		checkTreeNotEmpty();
		return root = impl.splay(DoubleBinarySearchTrees.findMax(root));
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> getPredecessor(DoubleObjReferenceableHeap.Ref <V> ref) {

	@Override
	public DoubleObjReferenceableHeap.Ref <V> getSuccessor(DoubleObjReferenceableHeap.Ref <V> ref) {
		SplayBSTNode <V> n = DoubleBinarySearchTrees.getSuccessor((SplayBSTNode <V>) ref);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public void meld(DoubleObjReferenceableHeap <V> heap) {
		Assertions.Heaps.noMeldWithSelf(this, heap);
		Assertions.Heaps.meldWithSameImpl(DoubleObjSplayTree.class, heap);
		Assertions.Heaps.equalComparatorBeforeMeld(comparator(), heap.comparator());
		DoubleObjSplayTree <V> h = (DoubleObjSplayTree <V>) heap;
		if (h.isEmpty())
			return;
		if (isEmpty()) {
			root = h.root;
			h.root = null;
			return;
		}

		DoubleComparator  cmp = c != null ? c : (a, b) -> Double.compare(a, b);

		double min1, max1, min2, max2;
		if (cmp.compare(max1 = findMax().key(), min2 = h.findMin().key()) <= 0) {
			/* all elements in this tree are <= than all elements in other tree */
			root = meld(this, h);
		} else if (cmp.compare(min1 = findMin().key(), max2 = h.findMax().key()) >= 0) {
			/* all elements in this tree are >= than all elements in other tree */
			root = meld(h, this);
		} else {
			int minCmp = cmp.compare(min1, min2);
			int maxCmp = cmp.compare(max1, max2);
			DoubleObjSplayTree <V> hLow = null, hHigh = null;

			if (minCmp < 0) {
				hLow = this.splitSmaller(min2);
			} else if (minCmp > 0) {
				hLow = h.splitSmaller(min1);
			}
			if (maxCmp < 0) {
				hHigh = h.splitGreater(max1);
			} else if (maxCmp > 0) {
				hHigh = this.splitGreater(max2);
			}

			if (h.root != null) {
				/* there is nothing smarter to do than 'addAll' for the shared range */
				/* We use 'insertNode' instead of 'insert' to maintain user references to nodes */
				for (SplayBSTNode <V> node = h.root;;) {
					for (;;) {
						while (node.hasLeftChild())
							node = node.left;
						if (!node.hasRightChild())
							break;
						node = node.right;
					}
					node.size = 1;
					SplayBSTNode <V> parent = node.parent;
					if (parent == null) {
						insertNode(node);
						break;
					} else {
						if (parent.right == node) {
							parent.right = null;
						} else {
							assert parent.left == node;
							parent.left = null;
						}
						node.parent = null;
						insertNode(node);
						node = parent;
					}
				}
			}

			if (hLow != null) {
				assert cmp.compare(hLow.findMax().key(), findMin().key()) < 0;
				root = meld(hLow, this);
			}
			if (hHigh != null) {
				assert cmp.compare(hHigh.findMin().key(), findMax().key()) > 0;
				root = meld(this, hHigh);
			}
		}
		h.root = null;
	}

	private static <V> SplayBSTNode <V> meld(DoubleObjSplayTree <V> t1, DoubleObjSplayTree <V> t2) {
		/* Assume all nodes in t1 are smaller than all nodes in t2 */

		/* Splay t1 max and t2 min */
		SplayBSTNode <V> n1 = (SplayBSTNode <V>) t1.findMax();
		assert n1.isRoot();
		assert !n1.hasRightChild();

		n1.right = t2.root;
		t2.root.parent = n1;
		n1.size += t2.root.size;
		return n1;
	}

	@Override
	public DoubleObjSplayTree <V> splitSmaller(double key) {
		DoubleObjSplayTree <V> newTree = new DoubleObjSplayTree <>(c);
		SplayBSTNode <V> pred = (SplayBSTNode <V>) findSmaller(key);
		if (pred == null)
			return newTree;
		assert pred.isRoot();

		if ((root = pred.right) != null) {
			pred.size -= pred.right.size;
			pred.right.parent = null;
			pred.right = null;
		}

		newTree.root = pred;
		return newTree;
	}

	@Override
	public DoubleObjSplayTree <V> splitGreater(double key) {
		SplayBSTNode <V> succ = (SplayBSTNode <V>) findGreater(key);
		if (succ == null)
			return new DoubleObjSplayTree <>(c);
		return split(succ);
	}

	@Override
	public DoubleObjSplayTree <V> split(DoubleObjReferenceableHeap.Ref <V> ref) {
		DoubleObjSplayTree <V> newTree = new DoubleObjSplayTree <>(c);

		SplayBSTNode <V> n = impl.splay((SplayBSTNode <V>) ref);
		assert n.isRoot();

		if ((root = n.left) != null) {
			n.size -= n.left.size;
			n.left.parent = null;
			n.left = null;
		}

		newTree.root = n;
		return newTree;
	}

	@Override
	public void clear() {
		if (root == null)
			return;
		DoubleBinarySearchTrees.clear(root);
		root = null;
	}

	static class BaseNode< N extends BaseNode< N>> extends DoubleBinarySearchTrees.Node< N> {

		BaseNode(double key) {
			super(key);
		}

		public double key() {
			return key;
		}

	}

	abstract static class SplayImpl< N extends BaseNode< N>> {

		SplayImpl() {}

		N splay(N n) {
			if (n == null || n.isRoot())
				return n;
			for (;;) {
				N parent = n.parent;

				if (parent.isRoot()) {
					/* zig */
					rotate(n);
					return n;
				}

				if (n.isLeftChild() == parent.isLeftChild()) {
					/* zig-zig */
					rotate(parent);
					rotate(n);
				} else {
					/* zig-zag */
					rotate(n);
					rotate(n);
				}

				if (n.isRoot())
					return n;
			}
		}

		private void rotate(N n) {
			N parent = n.parent, grandparent = parent.parent;

			beforeRotate(n);

			if (n.isLeftChild()) {

				parent.left = n.right;
				if (parent.hasLeftChild())
					parent.left.parent = parent;
				n.right = parent;

			} else {
				assert n.isRightChild();

				parent.right = n.left;
				if (parent.hasRightChild())
					parent.right.parent = parent;
				n.left = parent;
			}

			n.parent = grandparent;
			parent.parent = n;
			if (grandparent != null) {
				if (grandparent.left == parent)
					grandparent.left = n;
				else
					grandparent.right = n;
			}
		}

		void beforeRotate(N n) {}

	}

	private static class SplayBSTNode <V> extends BaseNode< SplayBSTNode <V>> implements DoubleObjReferenceableHeap.Ref <V> {

		// TODO consider removing the size field
		int size;

		private V value;

		SplayBSTNode(double key) {
			super(key);
			size = 1;
		}

		@Override
		public double key() {
			return key;
		}

		@Override
		public V value() {
			return value;
		}

		@Override
		public void setValue(V val) {
			value = val;
		}

		@Override
		public String toString() {
			return "{" + key + ":" + value + "}";
		}

		@Override
		void clearWithoutUserData() {
			super.clearWithoutUserData();
			size = 1;
		}

	}

	private static class SplayImplWithSize <V> extends SplayImpl< SplayBSTNode <V>> {

		@Override
		void beforeRotate(SplayBSTNode <V> n) {
			super.beforeRotate(n);

			SplayBSTNode <V> parent = n.parent;
			int parentOldSize = parent.size;

			if (n.isLeftChild()) {
				parent.size = parentOldSize - n.size + (n.hasRightChild() ? n.right.size : 0);
			} else {
				assert n.isRightChild();
				parent.size = parentOldSize - n.size + (n.hasLeftChild() ? n.left.size : 0);
			}

			n.size = parentOldSize;
		}

	}

	private void checkTreeNotEmpty() {
		if (root == null)
			throw new IllegalStateException("Tree is empty");
	}
