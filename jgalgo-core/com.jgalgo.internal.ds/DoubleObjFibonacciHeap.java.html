package com.jgalgo.internal.ds;

import java.util.Arrays;
import it.unimi.dsi.fastutil.doubles.DoubleComparator;
import java.util.Iterator;
import com.jgalgo.internal.util.Assertions;

/**
 * A Fibonacci heap implementation.
 *
 * <p>
 * A pointer based heap implementation that support almost any operation in \(O(1)\) amortized time, except
 * {@link #remove} which takes \(O(\log n)\) time amortized.
 *
 * <p>
 * Using this heap, Dijkstra's shortest-path algorithm can be implemented in time \(O(m + n \log n)\) rather than \(O(m
 * \log n)\) as the {@link #decreaseKey} operation is performed in \(O(1)\) time amortized.
 *
 * <p>
 * In practice, the Fibonacci heaps are quite complex, and in some cases is better to use a Oairing heap.
 *
 * @param  <V> the values type
 * @see        <a href="https://en.wikipedia.org/wiki/Fibonacci_heap">Wikipedia</a>
 * @author     Barak Ugav
 */
public class DoubleObjFibonacciHeap <V> implements DoubleObjReferenceableHeap <V> {

	private final DoubleComparator  c;
	private Node <V> minRoot;
	private Node <V> begin;
	private Node <V> end;
	private int size;

	@SuppressWarnings("unchecked")
	private Node <V>[] tempNodesArr = EmptyNodesArr;
	@SuppressWarnings("rawtypes")
	private static final Node[] EmptyNodesArr = new Node[0];

	/**
	 * Constructs a new, empty Fibonacci heap, ordered according to the natural ordering of its keys.
	 */
	public DoubleObjFibonacciHeap() {
		this(null);
	}

	/**
	 * Constructs a new, empty Fibonacci heap, with keys ordered according to the specified comparator.
	 *
	 * @param comparator the comparator that will be used to order this heap. If {@code null}, the
	 *                       {@linkplain Comparable natural ordering} of the keys will be used.
	 */
	public DoubleObjFibonacciHeap(DoubleComparator  comparator) {
		c = comparator;
	}

	@Override
	public DoubleComparator  comparator() {
		return c;
	}

	@Override
	public void clear() {
		for (Node <V> p = begin, next; p != null; p = next) {
			next = p.next;
			Trees.clear(p, n -> {
				n.value = null;
			});
		}

		begin = end = minRoot = null;
		size = 0;
	}

	@Override
	public boolean isEmpty() {
		return size == 0;
	}

	@Override
	public boolean isNotEmpty() {
		return size != 0;
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> insert(double key) {
		Node <V> n = new Node <>(key);
		if (minRoot != null) {
			Node <V> last = end;
			last.next = n;
			n.prev = last;
			compareToMinRoot(n);
		} else {
			begin = n;
			minRoot = n;
		}
		end = n;
		size++;
		return n;
	}

	@SuppressWarnings({ "unchecked", "rawtypes" })
	@Override
	public Iterator<DoubleObjReferenceableHeap.Ref <V>> iterator() {
		return (Iterator) new Trees.PreOrderIter<>(begin);
	}

	@Override
	public void meld(DoubleObjReferenceableHeap <V> heap) {
		Assertions.Heaps.noMeldWithSelf(this, heap);
		Assertions.Heaps.meldWithSameImpl(DoubleObjFibonacciHeap.class, heap);
		Assertions.Heaps.equalComparatorBeforeMeld(comparator(), heap.comparator());
		DoubleObjFibonacciHeap <V> h = (DoubleObjFibonacciHeap <V>) heap;
		if (h.isEmpty())
			return;

		if (size == 0) {
			minRoot = h.minRoot;
			begin = h.begin;
			end = h.end;
		} else {
			end.next = h.begin;
			h.begin.prev = end;
			end = h.end;
			compareToMinRoot(h.minRoot);
		}
		size += h.size;

		h.begin = h.end = h.minRoot = null;
		h.size = 0;
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> findMin() {
		Assertions.Heaps.notEmpty(this);
		return minRoot;
	}

	@Override
	public DoubleObjReferenceableHeap.Ref <V> extractMin() {
		DoubleObjReferenceableHeap.Ref <V> min = findMin();
		remove(min);
		return min;
	}

	private static <V> void cut(Node <V> p) {
		assert p.parent != null;
		Node <V> prev = p.prev;
		if (prev != null) {
			prev.next = p.next;
			p.prev = null;
		} else {
			assert p.parent.child == p;
			p.parent.child = p.next;
		}
		p.parent.degree--;
		if (p.next != null) {
			p.next.prev = prev;
			p.next = null;
		}
		p.parent = null;
		p.marked = false;
	}

	private void addRoot(Node <V> p) {
		Node <V> last = end;
		last.next = p;
		p.prev = last;
		end = p;
	}

	private void compareToMinRoot(Node <V> p) {
		assert p.parent == null;
		if (c == null) {
			if (minRoot.key > p.key)
				minRoot = p;
		} else {
			if (c.compare(minRoot.key, p.key) > 0)
				minRoot = p;
		}
	}

	private void mark(Node <V> p) {
		for (Node <V> q; p.parent != null; p = q) {
			if (!p.marked) {
				p.marked = true;
				break;
			}
			q = p.parent;
			cut(p);
			addRoot(p);
		}
	}

	@Override
	public void decreaseKey(DoubleObjReferenceableHeap.Ref <V> ref, double newKey) {
		Node <V> n = (Node <V>) ref;
		Assertions.Heaps.decreaseKeyIsSmaller(n.key, newKey, c);
		n.key = newKey;

		Node <V> parent = n.parent;
		if (parent == null) {
			compareToMinRoot(n);
			return;
		}
		if (c == null) {
			if (newKey >= parent.key)
				return;
		} else {
			if (c.compare(newKey, parent.key) >= 0)
				return;
		}
		cut(n);
		addRoot(n);
		compareToMinRoot(n);
		mark(parent);
	}

	@Override
	public void remove(DoubleObjReferenceableHeap.Ref <V> ref) {
		Node <V> prev, n = (Node <V>) ref;

		final boolean isMinRoot = n == minRoot;
		if (n.parent != null) {
			Node <V> parent = n.parent;
			cut(n);
			mark(parent);
		} else {
			if ((prev = n.prev) != null) {
				prev.next = n.next;
				n.prev = null;
			} else
				begin = n.next;
			if (n.next != null) {
				n.next.prev = prev;
				n.next = null;
			} else {
				end = prev;
			}
		}
		if (--size == 0) {
			minRoot = null;
			return;
		}

		// add n children
		Node <V> first = n.child, last = null;
		if (first != null) {
			for (Node <V> p = first;;) {
				p.parent = null;
				Node <V> next = p.next;
				if (next == null) {
					last = p;
					break;
				}
				p = next;
			}

			if (end != null) {
				end.next = first;
				first.prev = end;
			} else {
				/* root list is empty */
				begin = first;
			}
			end = last;
		}

		// union trees
		Node <V>[] newRoots = tempNodesArr;
		int maxDegree = getMaxDegree(size);
		if (newRoots.length < maxDegree) {
			@SuppressWarnings("unchecked")
			Node <V>[] t = new Node[Math.max(maxDegree, newRoots.length * 2)];
			newRoots = tempNodesArr = t;
		}

		if (c == null) {
			for (Node <V> next, p = begin; p != null; p = next) {
				next = p.next;

				int degree;
				for (Node <V> q; (q = newRoots[degree = p.degree]) != null;) {
					newRoots[degree] = null;
					p = unionDefaultCmp(p, q);
				}

				newRoots[degree] = p;
			}
		} else {
			for (Node <V> next, p = begin; p != null; p = next) {
				next = p.next;

				int degree;
				for (Node <V> q; (q = newRoots[degree = p.degree]) != null;) {
					newRoots[degree] = null;
					p = unionCustomCmp(p, q);
				}

				newRoots[degree] = p;
			}
		}
		prev = null;
		begin = null;
		for (Node <V> p : newRoots) {
			if (p == null)
				continue;
			if (prev == null)
				begin = p;
			else {
				prev.next = p;
			}
			p.prev = prev;
			prev = p;
		}
		end = prev;
		if (prev != null)
			prev.next = null;

		/* Find new minimum */
		if (isMinRoot) {
			Node <V> min = null;
			if (c == null) {
				for (Node <V> p : newRoots) {
					if (p == null)
						continue;
					if (min == null || min.key > p.key)
						min = p;
				}
			} else {
				for (Node <V> p : newRoots) {
					if (p == null)
						continue;
					if (min == null || c.compare(min.key, p.key) > 0)
						min = p;
				}
			}
			minRoot = min;
		}
		Arrays.fill(newRoots, 0, maxDegree, null);
	}

	private Node <V> unionDefaultCmp(Node <V> u, Node <V> v) {
		if (v == minRoot || u.key > v.key) {
			Node <V> temp = u;
			u = v;
			v = temp;
		}
		assert u.key <= v.key;
		addChild(u, v);
		return u;
	}

	private Node <V> unionCustomCmp(Node <V> u, Node <V> v) {
		if (v == minRoot || c.compare(u.key, v.key) > 0) {
			Node <V> temp = u;
			u = v;
			v = temp;
		}
		assert c.compare(u.key, v.key) <= 0;
		addChild(u, v);
		return u;
	}

	private static  <V> void addChild(Node <V> parent, Node <V> child) {
		child.parent = parent;
		child.prev = null;
		child.next = parent.child;
		if (parent.child != null)
			child.next.prev = child;
		parent.child = child;
		parent.degree++;
	}

	private static class Node <V> extends Trees.TreeNodeImpl<Node <V>> implements DoubleObjReferenceableHeap.Ref <V> {

		double key;
		V value;
		byte degree;
		boolean marked;

		Node(double key) {
			this.key = key;
		}

		@Override
		public double key() {
			return key;
		}

		@Override
		public V value() {
			return value;
		}

		@Override
		public void setValue(V val) {
			value = val;
		}

		@Override
		public String toString() {
			return "{" + key + ":" + value + "}";
		}

	}

	private static final double GOLDEN_RATION = (1 + Math.sqrt(5)) / 2;
	private static final double LOG_GOLDEN_RATION = Math.log(GOLDEN_RATION);
	private static final double LOG_GOLDEN_RATION_INV = 1 / LOG_GOLDEN_RATION;

	private static int getMaxDegree(int size) {
		return (int) (Math.log(size) * LOG_GOLDEN_RATION_INV) + 1;
	}

}
