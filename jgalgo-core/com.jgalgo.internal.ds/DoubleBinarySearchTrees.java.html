package com.jgalgo.internal.ds;

import it.unimi.dsi.fastutil.doubles.DoubleComparator;

class DoubleBinarySearchTrees {

	private DoubleBinarySearchTrees() {}

	static < NodeT extends Node< NodeT>> NodeT find(NodeT root, DoubleComparator  c, double key) {
		return findOrNeighbor(root, c, key, BinarySearchTrees.NeighborType.None);
	}

	static < NodeT extends Node< NodeT>> NodeT findOrSmaller(NodeT root, DoubleComparator  c, double key) {
		return findOrNeighbor(root, c, key, BinarySearchTrees.NeighborType.Predecessor);
	}

	static < NodeT extends Node< NodeT>> NodeT findOrGreater(NodeT root, DoubleComparator  c, double key) {
		return findOrNeighbor(root, c, key, BinarySearchTrees.NeighborType.Successor);
	}

	private static < NodeT extends Node< NodeT>> NodeT findOrNeighbor(NodeT root, DoubleComparator  c, double key,
			BinarySearchTrees.NeighborType neighborType) {
		if (root == null)
			return null;
		if (c == null) {
			for (NodeT p = root;;) {
				int cmp = Double.compare(key, p.key);
				if (cmp < 0) {
					if (!p.hasLeftChild())
						return neighborType.onLeftChildMissing(p);
					p = p.left;
				} else if (cmp > 0) {
					if (!p.hasRightChild())
						return neighborType.onRightChildMissing(p);
					p = p.right;
				} else {
					return p;
				}
			}
		} else {
			for (NodeT p = root;;) {
				int cmp = c.compare(key, p.key);
				if (cmp < 0) {
					if (!p.hasLeftChild())
						return neighborType.onLeftChildMissing(p);
					p = p.left;
				} else if (cmp > 0) {
					if (!p.hasRightChild())
						return neighborType.onRightChildMissing(p);
					p = p.right;
				} else {
					return p;
				}
			}
		}
	}

	static < NodeT extends Node< NodeT>> NodeT findSmaller(NodeT root, DoubleComparator  c, double key) {
		if (root == null)
			return null;
		if (c == null) {
			for (NodeT p = root;;) {
				int cmp = Double.compare(key, p.key);
				if (cmp <= 0) {
					if (!p.hasLeftChild())
						return BinarySearchTrees.getPredecessor(p);
					p = p.left;
				} else {
					if (!p.hasRightChild())
						return p;
					p = p.right;
				}
			}
		} else {
			for (NodeT p = root;;) {
				int cmp = c.compare(key, p.key);
				if (cmp <= 0) {
					if (!p.hasLeftChild())
						return BinarySearchTrees.getPredecessor(p);
					p = p.left;
				} else {
					if (!p.hasRightChild())
						return p;
					p = p.right;
				}
			}
		}
	}

	static < NodeT extends Node< NodeT>> NodeT findGreater(NodeT root, DoubleComparator  c, double key) {
		if (root == null)
			return null;
		if (c == null) {
			for (NodeT p = root;;) {
				int cmp = Double.compare(key, p.key);
				if (cmp >= 0) {
					if (!p.hasRightChild())
						return BinarySearchTrees.getSuccessor(p);
					p = p.right;
				} else {
					if (!p.hasLeftChild())
						return p;
					p = p.left;
				}
			}
		} else {
			for (NodeT p = root;;) {
				int cmp = c.compare(key, p.key);
				if (cmp >= 0) {
					if (!p.hasRightChild())
						return BinarySearchTrees.getSuccessor(p);
					p = p.right;
				} else {
					if (!p.hasLeftChild())
						return p;
					p = p.left;
				}
			}
		}
	}

	static < NodeT extends Node< NodeT>> void insert(NodeT root, DoubleComparator  c, NodeT n) {
		if (c == null) {
			for (NodeT parent = root;;) {
				int cmp = Double.compare(n.key, parent.key);
				if (cmp <= 0) {
					if (!parent.hasLeftChild()) {
						parent.left = n;
						n.parent = parent;
						return;
					}
					parent = parent.left;
				} else {
					if (!parent.hasRightChild()) {
						parent.right = n;
						n.parent = parent;
						return;
					}
					parent = parent.right;
				}
			}
		} else {
			for (NodeT parent = root;;) {
				int cmp = c.compare(n.key, parent.key);
				if (cmp <= 0) {
					if (!parent.hasLeftChild()) {
						parent.left = n;
						n.parent = parent;
						return;
					}
					parent = parent.left;
				} else {
					if (!parent.hasRightChild()) {
						parent.right = n;
						n.parent = parent;
						return;
					}
					parent = parent.right;
				}
			}
		}
	}

	static class Node< NodeT extends Node< NodeT>> extends BinarySearchTrees.Node<NodeT> {
		double key;

		Node(double key) {
			this.key = key;
		}


		@Override
		public String toString() {
			return String.valueOf(key);
		}
	}

}
