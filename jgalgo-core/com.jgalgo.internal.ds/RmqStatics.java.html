package com.jgalgo.internal.ds;

class RmqStatics {

	static void checkIndices(int i, int j, int n) {
		if (!(0 <= i && i <= j && j < n))
			throw new IllegalArgumentException("Illegal indices [" + i + "," + j + "]");
	}

}
