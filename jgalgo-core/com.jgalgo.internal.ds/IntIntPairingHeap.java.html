package com.jgalgo.internal.ds;

import java.util.Iterator;
import com.jgalgo.internal.util.Assertions;
import it.unimi.dsi.fastutil.Stack;
import it.unimi.dsi.fastutil.objects.ObjectArrayList;
import it.unimi.dsi.fastutil.ints.IntComparator;


/**
 * A Pairing heap implementation.
 *
 * <p>
 * A pointer based heap implementation that support almost any operation in \(O(1)\) amortized time, except
 * {@link #remove(IntIntReferenceableHeap.Ref)} and {@link #decreaseKey(IntIntReferenceableHeap.Ref, int)}
 * which takes \(O(\log n)\) time amortized.
 *
 * <p>
 * Using this heap, {@link ShortestPathSingleSourceDijkstra} can be implemented in time \(O(m + n \log n)\) rather than
 * \(O(m \log n)\) as the {@link #decreaseKey(IntIntReferenceableHeap.Ref, int)} operation is performed in
 * \(O(1)\) time amortized.
 *
 * <p>
 * Pairing heaps are one of the best pointer based heaps implementations in practice, and should be used as a default
 * choice for the general use case.
 *
 * @see    <a href="https://en.wikipedia.org/wiki/Pairing_heap">Wikipedia</a>
 * @author Barak Ugav
 */
public class IntIntPairingHeap  implements IntIntReferenceableHeap  {

	private final IntComparator  c;
	private Node  minRoot;

	public IntIntPairingHeap(IntComparator  c) {
		this.c = c;
	}

	@Override
	public IntComparator  comparator() {
		return c;
	}

	@Override
	public IntIntReferenceableHeap.Ref  findMin() {
		Assertions.Heaps.notEmpty(this);
		return minRoot;
	}

	@Override
	public boolean isEmpty() {
		return minRoot == null;
	}

	@Override
	public boolean isNotEmpty() {
		return minRoot != null;
	}

	private static  void cut(Node  n) {
		Node  next = n.next;
		if (next != null) {
			next.prevOrParent = n.prevOrParent;
			n.next = null;
		}
		if (n.prevOrParent.child == n) { /* n.parent.child == n */
			n.prevOrParent.child = next;
		} else {
			n.prevOrParent.next = next;
		}
		n.prevOrParent = null;
	}

	void addChild(Node  parent, Node  newChild) {
		assert newChild.prevOrParent == null;
		assert newChild.next == null;
		Node  oldChild = parent.child;
		if (oldChild != null) {
			oldChild.prevOrParent = newChild;
			newChild.next = oldChild;
		}
		parent.child = newChild;
		newChild.prevOrParent = parent;
	}

	void insertNode(Node  n) {
		if (minRoot == null) {
			minRoot = n;
		} else {
			minRoot = meld(minRoot, n);
		}
	}

	void afterKeyDecrease(Node  n) {
		if (n == minRoot)
			return;
		cut(n);
		minRoot = meld(minRoot, n);
	}

	@Override
	public IntIntReferenceableHeap.Ref  extractMin() {
		Node  min = minRoot;
		removeRoot();
		return min;
	}

	@Override
	public void remove(IntIntReferenceableHeap.Ref  ref) {
		Node  n = (Node ) ref;
		assert minRoot != null;
		if (n != minRoot) {
			cut(n);
			addChild(n, minRoot);
			minRoot = n;
		}
		removeRoot();
	}

	@Override
	public void meld(IntIntReferenceableHeap  heap) {
		Assertions.Heaps.noMeldWithSelf(this, heap);
		Assertions.Heaps.meldWithSameImpl(getClass(), heap);
		Assertions.Heaps.equalComparatorBeforeMeld(comparator(), heap.comparator());
		IntIntPairingHeap  h = (IntIntPairingHeap ) heap;

		if (minRoot == null) {
			minRoot = h.minRoot;
		} else if (h.minRoot != null) {
			minRoot = meld(minRoot, h.minRoot);
		}

		h.minRoot = null;
	}

	private Node  meld(Node  n1, Node  n2) {
		return c == null ? meldDefaultCmp(n1, n2) : meldCustomCmp(n1, n2);
	}

	Node  meldDefaultCmp(Node  n1, Node  n2) {
		assert n1.prevOrParent == null && n1.next == null;
		assert n2.prevOrParent == null && n2.next == null;

		/* assume n1 has smaller key than n2 */
		if (n1.key > n2.key) {
			Node  temp = n1;
			n1 = n2;
			n2 = temp;
		}

		addChild(n1, n2);
		return n1;
	}

	Node  meldCustomCmp(Node  n1, Node  n2) {
		assert n1.prevOrParent == null && n1.next == null;
		assert n2.prevOrParent == null && n2.next == null;

		/* assume n1 has smaller key than n2 */
		if (c.compare(n1.key, n2.key) > 0) {
			Node  temp = n1;
			n1 = n2;
			n2 = temp;
		}

		addChild(n1, n2);
		return n1;
	}

	@Override
	public void clear() {
		if (minRoot == null)
			return;

		for (Node  p = minRoot;;) {
			while (p.child != null) {
				p = p.child;
				while (p.next != null)
					p = p.next;
			}
			Node  prev = p.prevOrParent;
			if (prev == null)
				break;
			p.prevOrParent = null;
			if (prev.next == p) {
				prev.next = null;
			} else {
				prev.child = null;
			}
			p = prev;
		}

		minRoot = null;
	}

	@Override
	public Iterator<IntIntReferenceableHeap.Ref > iterator() {
		return new PreOrderIter (minRoot);
	}

	static class PreOrderIter  implements Iterator<IntIntReferenceableHeap.Ref > {

		private final Stack<Node > path = new ObjectArrayList<>();

		PreOrderIter(Node  p) {
			if (p != null)
				path.push(p);
		}

		@Override
		public boolean hasNext() {
			return !path.isEmpty();
		}

		@Override
		public Node  next() {
			Assertions.Iters.hasNext(this);
			final Node  ret = path.top();

			Node  next;
			if ((next = ret.child) != null) {
				path.push(next);
			} else {
				Node  p0;
				do {
					p0 = path.pop();
					if ((next = p0.next) != null) {
						path.push(next);
						break;
					}
				} while (!path.isEmpty());
			}

			return ret;
		}

	}

	@Override
	public IntIntReferenceableHeap.Ref  insert(int key) {
		Node  n = new Node (key);
		insertNode(n);
		return n;
	}


	private void removeRoot() {
		if (minRoot.child == null) {
			minRoot = null;
			return;
		}

		if (c == null) {
			/* meld pairs from left to right */
			Node  tail;
			for (Node  prev = minRoot, next = minRoot.child;;) {
				Node  n1 = next;
				if (n1 == null) {
					tail = prev;
					break;
				}

				Node  n2 = n1.next;
				if (n2 == null) {
					n1.prevOrParent = prev;
					tail = n1;
					break;
				}
				next = n2.next;
				n1.next = null;
				n2.next = null;
				n1.prevOrParent = null;
				n2.prevOrParent = null;
				n1 = meldDefaultCmp(n1, n2);

				n1.prevOrParent = prev;
				prev = n1;
			}
			minRoot.child = null;

			/* meld all from right to left */
			Node  root = tail, prev = root.prevOrParent;
			root.prevOrParent = null;
			for (;;) {
				Node  other = prev;
				if (other == minRoot) {
					minRoot = root;
					break;
				}
				prev = other.prevOrParent;
				other.prevOrParent = null;
				root = meldDefaultCmp(root, other);
			}
		} else {

			/* meld pairs from left to right */
			Node  tail;
			for (Node  prev = minRoot, next = minRoot.child;;) {
				Node  n1 = next;
				if (n1 == null) {
					tail = prev;
					break;
				}

				Node  n2 = n1.next;
				if (n2 == null) {
					n1.prevOrParent = prev;
					tail = n1;
					break;
				}
				next = n2.next;
				n1.next = null;
				n2.next = null;
				n1.prevOrParent = null;
				n2.prevOrParent = null;
				n1 = meldCustomCmp(n1, n2);

				n1.prevOrParent = prev;
				prev = n1;
			}
			minRoot.child = null;

			/* meld all from right to left */
			Node  root = tail, prev = root.prevOrParent;
			root.prevOrParent = null;
			for (;;) {
				Node  other = prev;
				if (other == minRoot) {
					minRoot = root;
					break;
				}
				prev = other.prevOrParent;
				other.prevOrParent = null;
				root = meldCustomCmp(root, other);
			}
		}
	}

	@Override
	public void decreaseKey(IntIntReferenceableHeap.Ref  ref, int newKey) {
		Node  n = (Node ) ref;
		Assertions.Heaps.decreaseKeyIsSmaller(n.key, newKey, c);
		n.key = newKey;
		afterKeyDecrease(n);
	}

	@Override
	public void increaseKey(IntIntReferenceableHeap.Ref  ref, int newKey) {
		Node  n = (Node ) ref;
		Assertions.Heaps.increaseKeyIsGreater(n.key, newKey, c);
		remove(n);
		n.key = newKey;
		insertNode(n);
	}

	private static class Node  implements IntIntReferenceableHeap.Ref  {
		int key;
		private int value;

		Node  prevOrParent;
		Node  next;
		Node  child;

		Node(int key) {
			this.key = key;
		}

		@Override
		public int key() {
			return key;
		}

		@Override
		public int value() {
			return value;
		}

		@Override
		public void setValue(int val) {
			value = val;
		}

		@Override
		public String toString() {
			return "{" + key + ":" + value + "}";
		}
	}

}
