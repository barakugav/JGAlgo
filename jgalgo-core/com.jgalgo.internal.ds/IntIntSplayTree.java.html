package com.jgalgo.internal.ds;

import it.unimi.dsi.fastutil.ints.IntComparator;

import java.util.Iterator;
import com.jgalgo.internal.util.Assertions;

/**
 * Splay binary search tree.
 *
 * <p>
 * The splay tree is a binary search tree which is not strictly balanced, in contrast to a red black tree. The splay
 * tree has a property that recent accessed elements are closer to the root, and therefore faster to access. All
 * operations have a time complexity of \(O(\log n)\) amortized time.
 *
 * <p>
 * The splay tree is specifically efficient for splits and joins operations.
 *
 * <p>
 * Based on 'Self-Adjusting Binary Search Trees' by Sleator and Tarjan (1985).
 *
 * @see        <a href="https://en.wikipedia.org/wiki/Splay_tree">Wikipedia</a>
 * @author     Barak Ugav
 */
public class IntIntSplayTree  implements IntIntBinarySearchTree  {

	private final IntComparator  c;
	private SplayBSTNode  root;
	private final SplayImpl< SplayBSTNode > impl = new SplayImpl<>();

	/**
	 * Constructs a new, empty splay tree, ordered according to the natural ordering of its keys.
	 */
	public IntIntSplayTree() {
		this(null);
	}

	/**
	 * Constructs a new, empty splay tree, with keys ordered according to the specified comparator.
	 *
	 * @param comparator the comparator that will be used to order this tree. If {@code null}, the
	 *                       {@linkplain Comparable natural ordering} of the keys will be used.
	 */
	public IntIntSplayTree(IntComparator  comparator) {
		this.c = comparator;
	}

	@Override
	public boolean isEmpty() {
		return root == null;
	}

	@Override
	public boolean isNotEmpty() {
		return root != null;
	}

	@Override
	public IntComparator  comparator() {
		return c;
	}

	@Override
	public IntIntReferenceableHeap.Ref  extractMin() {
		IntIntReferenceableHeap.Ref  min = findMin();
		remove(min);
		return min;
	}

	@Override
	public IntIntReferenceableHeap.Ref  extractMax() {
		IntIntReferenceableHeap.Ref  max = findMax();
		remove(max);
		return max;
	}

	@SuppressWarnings({ "unchecked", "rawtypes" })
	@Override
	public Iterator<IntIntReferenceableHeap.Ref > iterator() {
		return (Iterator) new IntBinarySearchTrees.BSTIterator <>(root);
	}

	@Override
	public IntIntReferenceableHeap.Ref  insert(int key) {
		return insertNode(new SplayBSTNode (key));
	}

	private SplayBSTNode  insertNode(SplayBSTNode  n) {
		assert n.parent == null;
		assert n.left == null;
		assert n.right == null;
		if (root == null)
			return root = n;

		IntBinarySearchTrees.insert(root, c, n);
		return root = impl.splay(n);
	}

	@Override
	public void remove(IntIntReferenceableHeap.Ref  ref) {
		SplayBSTNode  n = (SplayBSTNode ) ref;

		/* If the node has two children, swap with successor */
		if (n.hasLeftChild() && n.hasRightChild())
			IntBinarySearchTrees.swap(n, IntBinarySearchTrees.getSuccessor(n));

		SplayBSTNode  child;
		if (!n.hasLeftChild()) {
			replace(n, child = n.right);
		} else {
			assert !n.hasRightChild();
			replace(n, child = n.left);
		}


		SplayBSTNode  parent = n.parent;
		n.clearWithoutUserData();
		root = impl.splay(child != null ? child : parent);
	}

	private static  void replace(SplayBSTNode  u, SplayBSTNode  v) {
		/* replaces u with v */
		if (u.parent != null) {
			if (u.isLeftChild()) {
				u.parent.left = v;
			} else {
				assert u.isRightChild();
				u.parent.right = v;
			}
		}
		if (v != null)
			v.parent = u.parent;
	}

	@Override
	public void decreaseKey(IntIntReferenceableHeap.Ref  ref, int newKey) {
		SplayBSTNode  n = (SplayBSTNode ) ref;
		Assertions.Heaps.decreaseKeyIsSmaller(n.key, newKey, c);
		remove(n);
		n.key = newKey;
		insertNode(n);
	}

	@Override
	public void increaseKey(IntIntReferenceableHeap.Ref  ref, int newKey) {
		SplayBSTNode  n = (SplayBSTNode ) ref;
		Assertions.Heaps.increaseKeyIsGreater(n.key, newKey, c);
		remove(n);
		n.key = newKey;
		insertNode(n);
	}

	@Override
	public IntIntReferenceableHeap.Ref  find(int key) {
		SplayBSTNode  n = IntBinarySearchTrees.find(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public IntIntReferenceableHeap.Ref  findMin() {
		checkTreeNotEmpty();
		return root = impl.splay(IntBinarySearchTrees.findMin(root));
	}

	@Override
	public IntIntReferenceableHeap.Ref  findMax() {
		checkTreeNotEmpty();
		return root = impl.splay(IntBinarySearchTrees.findMax(root));
	}

	@Override
	public IntIntReferenceableHeap.Ref  findOrSmaller(int key) {
		SplayBSTNode  n = IntBinarySearchTrees.findOrSmaller(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public IntIntReferenceableHeap.Ref  findOrGreater(int key) {
		SplayBSTNode  n = IntBinarySearchTrees.findOrGreater(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public IntIntReferenceableHeap.Ref  findSmaller(int key) {
		SplayBSTNode  n = IntBinarySearchTrees.findSmaller(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public IntIntReferenceableHeap.Ref  findGreater(int key) {
		SplayBSTNode  n = IntBinarySearchTrees.findGreater(root, c, key);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public IntIntReferenceableHeap.Ref  getPredecessor(IntIntReferenceableHeap.Ref  ref) {
		SplayBSTNode  n = IntBinarySearchTrees.getPredecessor((SplayBSTNode ) ref);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public IntIntReferenceableHeap.Ref  getSuccessor(IntIntReferenceableHeap.Ref  ref) {
		SplayBSTNode  n = IntBinarySearchTrees.getSuccessor((SplayBSTNode ) ref);
		return n == null ? null : (root = impl.splay(n));
	}

	@Override
	public void meld(IntIntReferenceableHeap  heap) {
		Assertions.Heaps.noMeldWithSelf(this, heap);
		Assertions.Heaps.meldWithSameImpl(IntIntSplayTree.class, heap);
		Assertions.Heaps.equalComparatorBeforeMeld(comparator(), heap.comparator());
		IntIntSplayTree  h = (IntIntSplayTree ) heap;
		if (h.isEmpty())
			return;
		if (isEmpty()) {
			root = h.root;
			h.root = null;
			return;
		}

		IntComparator  cmp = c != null ? c : (a, b) -> Integer.compare(a, b);

		int min1, max1, min2, max2;
		if (cmp.compare(max1 = findMax().key(), min2 = h.findMin().key()) <= 0) {
			/* all elements in this tree are <= than all elements in other tree */
			root = meld(this, h);
		} else if (cmp.compare(min1 = findMin().key(), max2 = h.findMax().key()) >= 0) {
			/* all elements in this tree are >= than all elements in other tree */
			root = meld(h, this);
		} else {
			int minCmp = cmp.compare(min1, min2);
			int maxCmp = cmp.compare(max1, max2);
			IntIntSplayTree  hLow = null, hHigh = null;

			if (minCmp < 0) {
				hLow = this.splitSmaller(min2);
			} else if (minCmp > 0) {
				hLow = h.splitSmaller(min1);
			}
			if (maxCmp < 0) {
				hHigh = h.splitGreater(max1);
			} else if (maxCmp > 0) {
				hHigh = this.splitGreater(max2);
			}

			if (h.root != null) {
				/* there is nothing smarter to do than 'addAll' for the shared range */
				/* We use 'insertNode' instead of 'insert' to maintain user references to nodes */
				for (SplayBSTNode  node = h.root;;) {
					for (;;) {
						while (node.hasLeftChild())
							node = node.left;
						if (!node.hasRightChild())
							break;
						node = node.right;
					}
					SplayBSTNode  parent = node.parent;
					if (parent == null) {
						insertNode(node);
						break;
					} else {
						if (parent.right == node) {
							parent.right = null;
						} else {
							assert parent.left == node;
							parent.left = null;
						}
						node.parent = null;
						insertNode(node);
						node = parent;
					}
				}
			}

			if (hLow != null) {
				assert cmp.compare(hLow.findMax().key(), findMin().key()) < 0;
				root = meld(hLow, this);
			}
			if (hHigh != null) {
				assert cmp.compare(hHigh.findMin().key(), findMax().key()) > 0;
				root = meld(this, hHigh);
			}
		}
		h.root = null;
	}

	private static  SplayBSTNode  meld(IntIntSplayTree  t1, IntIntSplayTree  t2) {
		/* Assume all nodes in t1 are smaller than all nodes in t2 */

		/* Splay t1 max and t2 min */
		SplayBSTNode  n1 = (SplayBSTNode ) t1.findMax();
		assert n1.isRoot();
		assert !n1.hasRightChild();

		n1.right = t2.root;
		t2.root.parent = n1;
		return n1;
	}

	@Override
	public IntIntSplayTree  splitSmaller(int key) {
		IntIntSplayTree  newTree = new IntIntSplayTree (c);
		SplayBSTNode  pred = (SplayBSTNode ) findSmaller(key);
		if (pred == null)
			return newTree;
		assert pred.isRoot();

		if ((root = pred.right) != null) {
			pred.right.parent = null;
			pred.right = null;
		}

		newTree.root = pred;
		return newTree;
	}

	@Override
	public IntIntSplayTree  splitGreater(int key) {
		SplayBSTNode  succ = (SplayBSTNode ) findGreater(key);
		if (succ == null)
			return new IntIntSplayTree (c);
		return split(succ);
	}

	@Override
	public IntIntSplayTree  split(IntIntReferenceableHeap.Ref  ref) {
		IntIntSplayTree  newTree = new IntIntSplayTree (c);

		SplayBSTNode  n = impl.splay((SplayBSTNode ) ref);
		assert n.isRoot();

		if ((root = n.left) != null) {
			n.left.parent = null;
			n.left = null;
		}

		newTree.root = n;
		return newTree;
	}

	@Override
	public void clear() {
		if (root == null)
			return;
		IntBinarySearchTrees.clear(root);
		root = null;
	}

	static class BaseNode< N extends BaseNode< N>> extends IntBinarySearchTrees.Node< N> {

		BaseNode(int key) {
			super(key);
		}

		public int key() {
			return key;
		}

	}

	static class SplayImpl< N extends BaseNode< N>> {

		SplayImpl() {}

		N splay(N n) {
			if (n == null || n.isRoot())
				return n;
			for (;;) {
				N parent = n.parent;

				if (parent.isRoot()) {
					/* zig */
					rotate(n);
					return n;
				}

				if (n.isLeftChild() == parent.isLeftChild()) {
					/* zig-zig */
					rotate(parent);
					rotate(n);
				} else {
					/* zig-zag */
					rotate(n);
					rotate(n);
				}

				if (n.isRoot())
					return n;
			}
		}

		private void rotate(N n) {
			N parent = n.parent, grandparent = parent.parent;

			beforeRotate(n);

			if (n.isLeftChild()) {

				parent.left = n.right;
				if (parent.hasLeftChild())
					parent.left.parent = parent;
				n.right = parent;

			} else {
				assert n.isRightChild();

				parent.right = n.left;
				if (parent.hasRightChild())
					parent.right.parent = parent;
				n.left = parent;
			}

			n.parent = grandparent;
			parent.parent = n;
			if (grandparent != null) {
				if (grandparent.left == parent)
					grandparent.left = n;
				else
					grandparent.right = n;
			}
		}

		void beforeRotate(N n) {}

	}

	private static class SplayBSTNode  extends BaseNode< SplayBSTNode > implements IntIntReferenceableHeap.Ref  {

		private int value;

		SplayBSTNode(int key) {
			super(key);
		}

		@Override
		public int key() {
			return key;
		}

		@Override
		public int value() {
			return value;
		}

		@Override
		public void setValue(int val) {
			value = val;
		}

		@Override
		public String toString() {
			return "{" + key + ":" + value + "}";
		}

	}

	private void checkTreeNotEmpty() {
		if (root == null)
			throw new IllegalStateException("Tree is empty");
	}

}
