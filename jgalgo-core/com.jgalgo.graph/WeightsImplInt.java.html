package com.jgalgo.graph;

import static com.jgalgo.internal.util.Range.range;
import java.util.Arrays;
import it.unimi.dsi.fastutil.ints.IntArrays;
import it.unimi.dsi.fastutil.ints.IntSet;

class WeightsImplInt {

	private WeightsImplInt() {}

	abstract static class IndexImpl  extends WeightsImpl.IndexAbstract<Integer> implements IWeightsInt  {

		int[] weights;
		final int defaultWeight;


		IndexImpl(IntSet elements, boolean isEdges, int defVal) {
			super(elements, isEdges);

			defaultWeight = defVal;

			weights = IntArrays.EMPTY_ARRAY;
			if (!elements.isEmpty()) {
				weights = Arrays.copyOf(weights, elements.size());
				Arrays.fill(weights, defaultWeight);
			}
		}

		IndexImpl(WeightsImplInt.IndexImpl  orig, IntSet elements, boolean isEdges) {
			this(elements, isEdges, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(orig.weights, elements.size());
		}

		IndexImpl(WeightsImplInt.IndexImpl  orig, IntSet elements, boolean isEdges, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			this(elements, isEdges, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(weights, elements.size());
			for (int i : range(elements.size()))
				weights[reIndexMap.origToReIndexed(i)] = orig.weights[i];
		}

		
		@Override
		public int get(int idx) {
			checkIdx(idx);
			return  weights[idx];
		}

		@Override
		public int defaultWeight() {
			return defaultWeight;
		}

	}

	static class IndexImmutable  extends WeightsImplInt.IndexImpl  implements WeightsImpl.IndexImmutable<Integer> {

		IndexImmutable(WeightsImplInt.IndexImpl  orig, IntSet elements, boolean isEdges) {
			super(orig, elements, isEdges);
		}

		IndexImmutable(WeightsImplInt.IndexImpl  orig, IntSet elements, boolean isEdges, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(orig, elements, isEdges, reIndexMap);
		}

		@Override
		public void set(int idx, int weight) {
			throw new UnsupportedOperationException("immutable weights");
		}
	}

	static class IndexMutable  extends WeightsImplInt.IndexImpl  implements WeightsImpl.IndexMutable<Integer> {

		IndexMutable(IntSet elements, boolean isEdges, int defVal) {
			super(elements, isEdges, defVal);
		}

		IndexMutable(WeightsImplInt.IndexImpl  orig, IntSet elements, boolean isEdges) {
			super(orig, elements, isEdges);
		}

		@Override
		public void set(int idx, int weight) {
			checkIdx(idx);
			weights[idx] = weight;
		}


		@Override
		public int capacity() {
			return weights.length;
		}

		@Override
		public void expand(int newCapacity) {
			int oldCapacity = capacity();
			assert oldCapacity < newCapacity;
			weights = Arrays.copyOf(weights, newCapacity);
			Arrays.fill(weights, oldCapacity, newCapacity, defaultWeight);
		}

		@Override
		public void swapAndClear(int removedIdx, int swappedIdx) {
			// checkIdx(removedIdx);
			// checkIdx(swappedIdx);
			weights[removedIdx] = weights[swappedIdx];
			weights[swappedIdx] = defaultWeight;
		}

		@Override
		public void clear(int idx) {
			weights[idx] = defaultWeight;
		}

		@Override
		public void clear() {
			Arrays.fill(weights, 0, size(), defaultWeight);
		}
	}

	static class IntMapped  extends WeightsImpl.IntMapped<Integer> implements IWeightsInt  {
		IntMapped(WeightsImplInt.IndexImpl  weights, IndexIntIdMap indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplInt.IndexImpl  weights() {
			return (WeightsImplInt.IndexImpl ) super.weights();
		}

		@Override
		public int get(int id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(int id, int weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public int defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjMapped<K > extends WeightsImpl.ObjMapped<K, Integer> implements WeightsInt<K > {
		ObjMapped(WeightsImplInt.IndexImpl  weights, IndexIdMap<K> indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplInt.IndexImpl  weights() {
			return (WeightsImplInt.IndexImpl ) super.weights();
		}

		@Override
		public int get(K id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(K id, int weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public int defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class IntImmutableView  extends WeightsImpl.IntImmutableView<Integer> implements IWeightsInt  {
		IntImmutableView(IWeightsInt  w) {
			super(w);
		}

		@Override
		IWeightsInt  weights() {
			return (IWeightsInt ) super.weights();
		}

		@Override
		public int get(int id) {
			return weights().get(id);
		}

		@Override
		public void set(int id, int weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public int defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjImmutableView<K > extends WeightsImpl.ObjImmutableView<K, Integer> implements WeightsInt <K > {
		ObjImmutableView(WeightsInt <K > w) {
			super(w);
		}

		@Override
		WeightsInt <K > weights() {
			return (WeightsInt <K >) super.weights();
		}

		@Override
		public int get(K id) {
			return weights().get(id);
		}

		@Override
		public void set(K id, int weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public int defaultWeight() {
			return weights().defaultWeight();
		}
	}

}
