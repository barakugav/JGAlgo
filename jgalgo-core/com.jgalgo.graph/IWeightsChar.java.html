package com.jgalgo.graph;

/**
 * Specific weights of {@code char} for {@linkplain IntGraph int graphs}.
 *
 * @author Barak Ugav
 */
public interface IWeightsChar  extends IWeights<Character>, WeightsChar<Integer >  {

	/**
	 * Get the weight associated with the given element.
	 *
	 * @param  element an element (edge/vertex.)
	 * @return    the weight associated with the given element.
	 */
	public char get(int element);

	@Deprecated
	@Override
	default char get(Integer element) {
		return get(element.intValue());
	}

	@Deprecated
	@Override
	default Character getAsObj(int element) {
		return Character.valueOf(get(element));
	}

	/**
	 * Set the weight associated with the given element.
	 *
	 * @param element     an element (edge/vertex)
	 * @param weight new weight that will be associated with the given element
	 */
	public void set(int element, char weight);

	@Deprecated
	@Override
	default void set(Integer element, char weight) {
		set(element.intValue(), weight);
	}

	@Deprecated
	@Override
	default void setAsObj(int element, Character weight) {
		set(element, weight.charValue());
	}

	@Deprecated
	@Override
	default Character getAsObj(Integer element) {
		return Character.valueOf(get(element.intValue()));
	}

	@Deprecated
	@Override
	default void setAsObj(Integer element, Character weight) {
		set(element.intValue(), weight.charValue());
	}

	

}
