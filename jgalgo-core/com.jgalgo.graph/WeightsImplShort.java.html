package com.jgalgo.graph;

import java.util.Arrays;
import it.unimi.dsi.fastutil.shorts.ShortArrays;

class WeightsImplShort {

	abstract static class IndexImpl  extends WeightsImpl.IndexAbstract<Short> implements IWeightsShort  {

		short[] weights;
		final short defaultWeight;

		IndexImpl(GraphElementSet elements, short defVal) {
			super(elements);

			weights = ShortArrays.EMPTY_ARRAY;
			defaultWeight = defVal;
		}

		IndexImpl(WeightsImplShort.IndexImpl  orig, GraphElementSet elements) {
			this(elements, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(orig.weights, elements.size());
		}

		IndexImpl(WeightsImplShort.IndexImpl  orig, GraphElementSet elements,
				IndexGraphBuilder.ReIndexingMap reIndexMap) {
			this(elements, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(weights, elements.size());
			final int s = elements.size();
			for (int i = 0; i < s; i++)
				weights[reIndexMap.origToReIndexed(i)] = orig.weights[i];
		}

		
		@Override
		public short get(int idx) {
			checkIdx(idx);
			return  weights[idx];
		}

		@Override
		public short defaultWeight() {
			return defaultWeight;
		}
	}

	static class IndexImmutable  extends WeightsImplShort.IndexImpl  implements WeightsImpl.IndexImmutable<Short> {

		IndexImmutable(WeightsImplShort.IndexImpl  orig, GraphElementSet.FixedSize elements) {
			super(orig, elements);
		}

		IndexImmutable(WeightsImplShort.IndexImpl  orig, GraphElementSet.FixedSize elements,
				IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(orig, elements, reIndexMap);
		}

		@Override
		public void set(int idx, short weight) {
			throw new UnsupportedOperationException("immutable weights");
		}
	}

	static class IndexMutable  extends WeightsImplShort.IndexImpl  implements WeightsImpl.IndexMutable<Short> {

		IndexMutable(GraphElementSet elements, short defVal) {
			super(elements, defVal);
		}

		IndexMutable(WeightsImplShort.IndexImpl  orig, GraphElementSet elements) {
			super(orig, elements);
		}

		@Override
		public void set(int idx, short weight) {
			checkIdx(idx);
			weights[idx] = weight;
		}

		@Override
		public int capacity() {
			return weights.length;
		}

		@Override
		public void expand(int newCapacity) {
			int oldCapacity = weights.length;
			assert oldCapacity < newCapacity;
			weights = Arrays.copyOf(weights, newCapacity);
			Arrays.fill(weights, oldCapacity, newCapacity, defaultWeight);
		}

		@Override
		public void swapAndClear(int removedIdx, int swappedIdx) {
			// checkIdx(removedIdx);
			// checkIdx(swappedIdx);
			weights[removedIdx] = weights[swappedIdx];
			weights[swappedIdx] = defaultWeight;
		}

		@Override
		public void clear(int idx) {
			weights[idx] = defaultWeight;
		}

		@Override
		public void clear() {
			Arrays.fill(weights, 0, size(), defaultWeight);
		}
	}

	static class IntMapped  extends WeightsImpl.IntMapped<Short> implements IWeightsShort  {
		IntMapped(WeightsImplShort.IndexImpl  weights, IndexIntIdMap indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplShort.IndexImpl  weights() {
			return (WeightsImplShort.IndexImpl ) super.weights();
		}

		@Override
		public short get(int id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(int id, short weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public short defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjMapped<K > extends WeightsImpl.ObjMapped<K, Short> implements WeightsShort<K > {
		ObjMapped(WeightsImplShort.IndexImpl  weights, IndexIdMap<K> indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplShort.IndexImpl  weights() {
			return (WeightsImplShort.IndexImpl ) super.weights();
		}

		@Override
		public short get(K id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(K id, short weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public short defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class IntImmutableView  extends WeightsImpl.IntImmutableView<Short> implements IWeightsShort  {
		IntImmutableView(IWeightsShort  w) {
			super(w);
		}

		@Override
		IWeightsShort  weights() {
			return (IWeightsShort ) super.weights();
		}

		@Override
		public short get(int id) {
			return weights().get(id);
		}

		@Override
		public void set(int id, short weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public short defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjImmutableView<K > extends WeightsImpl.ObjImmutableView<K, Short> implements WeightsShort <K > {
		ObjImmutableView(WeightsShort <K > w) {
			super(w);
		}

		@Override
		WeightsShort <K > weights() {
			return (WeightsShort <K >) super.weights();
		}

		@Override
		public short get(K id) {
			return weights().get(id);
		}

		@Override
		public void set(K id, short weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public short defaultWeight() {
			return weights().defaultWeight();
		}
	}

}
