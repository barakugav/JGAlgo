package com.jgalgo.graph;

/**
 * Specific weights of {@code Object} for {@linkplain IntGraph int graphs}.
 *
 * @author Barak Ugav
 */
public interface IWeightsObj <T> extends IWeights<T>, WeightsObj<Integer , T>  {

	/**
	 * Get the weight associated with the given element.
	 *
	 * @param  element an element (edge/vertex.)
	 * @return    the weight associated with the given element.
	 */
	public T get(int element);

	@Deprecated
	@Override
	default T get(Integer element) {
		return get(element.intValue());
	}

	@Deprecated
	@Override
	default T getAsObj(int element) {
		return get(element);
	}

	/**
	 * Set the weight associated with the given element.
	 *
	 * @param element     an element (edge/vertex)
	 * @param weight new weight that will be associated with the given element
	 */
	public void set(int element, T weight);

	@Deprecated
	@Override
	default void set(Integer element, T weight) {
		set(element.intValue(), weight);
	}

	@Deprecated
	@Override
	default void setAsObj(int element, T weight) {
		set(element, weight);
	}

	@Deprecated
	@Override
	default T getAsObj(Integer element) {
		return get(element.intValue());
	}

	@Deprecated
	@Override
	default void setAsObj(Integer element, T weight) {
		set(element.intValue(), weight);
	}

	

}
