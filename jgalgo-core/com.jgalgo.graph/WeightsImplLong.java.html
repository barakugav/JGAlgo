package com.jgalgo.graph;

import java.util.Arrays;
import it.unimi.dsi.fastutil.longs.LongArrays;
import it.unimi.dsi.fastutil.ints.IntSet;

class WeightsImplLong {

	private WeightsImplLong() {}

	abstract static class IndexImpl  extends WeightsImpl.IndexAbstract<Long> implements IWeightsLong  {

		long[] weights;
		final long defaultWeight;

		IndexImpl(IntSet elements, boolean isEdges, long defVal) {
			super(elements, isEdges);

			weights = LongArrays.EMPTY_ARRAY;
			defaultWeight = defVal;
		}

		IndexImpl(WeightsImplLong.IndexImpl  orig, IntSet elements, boolean isEdges) {
			this(elements, isEdges, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(orig.weights, elements.size());
		}

		IndexImpl(WeightsImplLong.IndexImpl  orig, IntSet elements, boolean isEdges, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			this(elements, isEdges, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(weights, elements.size());
			final int s = elements.size();
			for (int i = 0; i < s; i++)
				weights[reIndexMap.origToReIndexed(i)] = orig.weights[i];
		}

		
		@Override
		public long get(int idx) {
			checkIdx(idx);
			return  weights[idx];
		}

		@Override
		public long defaultWeight() {
			return defaultWeight;
		}
	}

	static class IndexImmutable  extends WeightsImplLong.IndexImpl  implements WeightsImpl.IndexImmutable<Long> {

		IndexImmutable(WeightsImplLong.IndexImpl  orig, IntSet elements, boolean isEdges) {
			super(orig, elements, isEdges);
		}

		IndexImmutable(WeightsImplLong.IndexImpl  orig, IntSet elements, boolean isEdges, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(orig, elements, isEdges, reIndexMap);
		}

		@Override
		public void set(int idx, long weight) {
			throw new UnsupportedOperationException("immutable weights");
		}
	}

	static class IndexMutable  extends WeightsImplLong.IndexImpl  implements WeightsImpl.IndexMutable<Long> {

		IndexMutable(IntSet elements, boolean isEdges, long defVal) {
			super(elements, isEdges, defVal);
		}

		IndexMutable(WeightsImplLong.IndexImpl  orig, IntSet elements, boolean isEdges) {
			super(orig, elements, isEdges);
		}

		@Override
		public void set(int idx, long weight) {
			checkIdx(idx);
			weights[idx] = weight;
		}

		@Override
		public int capacity() {
			return weights.length;
		}

		@Override
		public void expand(int newCapacity) {
			int oldCapacity = weights.length;
			assert oldCapacity < newCapacity;
			weights = Arrays.copyOf(weights, newCapacity);
			Arrays.fill(weights, oldCapacity, newCapacity, defaultWeight);
		}

		@Override
		public void swapAndClear(int removedIdx, int swappedIdx) {
			// checkIdx(removedIdx);
			// checkIdx(swappedIdx);
			weights[removedIdx] = weights[swappedIdx];
			weights[swappedIdx] = defaultWeight;
		}

		@Override
		public void clear(int idx) {
			weights[idx] = defaultWeight;
		}

		@Override
		public void clear() {
			Arrays.fill(weights, 0, size(), defaultWeight);
		}
	}

	static class IntMapped  extends WeightsImpl.IntMapped<Long> implements IWeightsLong  {
		IntMapped(WeightsImplLong.IndexImpl  weights, IndexIntIdMap indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplLong.IndexImpl  weights() {
			return (WeightsImplLong.IndexImpl ) super.weights();
		}

		@Override
		public long get(int id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(int id, long weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public long defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjMapped<K > extends WeightsImpl.ObjMapped<K, Long> implements WeightsLong<K > {
		ObjMapped(WeightsImplLong.IndexImpl  weights, IndexIdMap<K> indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplLong.IndexImpl  weights() {
			return (WeightsImplLong.IndexImpl ) super.weights();
		}

		@Override
		public long get(K id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(K id, long weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public long defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class IntImmutableView  extends WeightsImpl.IntImmutableView<Long> implements IWeightsLong  {
		IntImmutableView(IWeightsLong  w) {
			super(w);
		}

		@Override
		IWeightsLong  weights() {
			return (IWeightsLong ) super.weights();
		}

		@Override
		public long get(int id) {
			return weights().get(id);
		}

		@Override
		public void set(int id, long weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public long defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjImmutableView<K > extends WeightsImpl.ObjImmutableView<K, Long> implements WeightsLong <K > {
		ObjImmutableView(WeightsLong <K > w) {
			super(w);
		}

		@Override
		WeightsLong <K > weights() {
			return (WeightsLong <K >) super.weights();
		}

		@Override
		public long get(K id) {
			return weights().get(id);
		}

		@Override
		public void set(K id, long weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public long defaultWeight() {
			return weights().defaultWeight();
		}
	}

}
