package com.jgalgo.graph;

import static com.jgalgo.internal.util.Range.range;
import java.util.Arrays;
import it.unimi.dsi.fastutil.longs.LongArrays;
import it.unimi.dsi.fastutil.ints.IntSet;

class WeightsImplBool {

	private WeightsImplBool() {}

	abstract static class IndexImpl  extends WeightsImpl.IndexAbstract<Boolean> implements IWeightsBool  {

		long[] weights;
		final boolean defaultWeight;

		static final int WordSize = Long.SIZE;

		IndexImpl(IntSet elements, boolean isVertices, boolean defVal) {
			super(elements, isVertices);
			defaultWeight = defVal;
			weights = LongArrays.EMPTY_ARRAY;
			if (!elements.isEmpty()) {
				weights = new long[wordsNum(elements.size())];
				if (defaultWeight)
					Arrays.fill(weights, ~0L);
			}
		}

		IndexImpl(IWeightsBool  orig, IntSet elements, boolean isVertices) {
			super(elements, isVertices);
			defaultWeight = orig.defaultWeight();
			final int n = elements.size();

			if (orig instanceof WeightsImplBool.IndexImpl) {
				WeightsImplBool.IndexImpl  orig0 = (WeightsImplBool.IndexImpl ) orig;
				WeightsImpl.checkSameSize(elements, orig0.elements);
				weights = Arrays.copyOf(orig0.weights, wordsNum(n));
			} else {
				weights = LongArrays.EMPTY_ARRAY;
				if (n > 0) {
					weights = new long[wordsNum(n)];
					if (defaultWeight)
						weights[weights.length - 1] = ~0L;
					for (int i : range(n))
						setImpl(i, orig.get(i));
				}
			}
		}

		IndexImpl(IWeightsBool  orig, IntSet elements, boolean isVertices, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(elements, isVertices);
			defaultWeight = orig.defaultWeight();
			final int n = elements.size();
			weights = new long[wordsNum(n)];
			if (defaultWeight)
				Arrays.fill(weights, ~0L);

			if (orig instanceof WeightsImplBool.IndexImpl) {
				WeightsImplBool.IndexImpl  orig0 = (WeightsImplBool.IndexImpl ) orig;
				if (defaultWeight) {
					for (int oldIdx : range(n))
						if ((orig0.weights[word(oldIdx)] & bit(oldIdx)) == 0)
							setImpl(reIndexMap.map(oldIdx), false);
				} else {
					for (int oldIdx : range(n))
						if ((orig0.weights[word(oldIdx)] & bit(oldIdx)) != 0)
							setImpl(reIndexMap.map(oldIdx), true);
				}
			} else {
				if (defaultWeight) {
					for (int oldIdx : range(n))
						if (!orig.get(oldIdx))
							setImpl(reIndexMap.map(oldIdx), false);
				} else {
					for (int oldIdx : range(n))
						if (orig.get(oldIdx))
							setImpl(reIndexMap.map(oldIdx), true);
				}
			}
		}

		
		@Override
		public boolean get(int idx) {
			checkIdx(idx);
			return (weights[word(idx)] & bit(idx)) != 0;
		}

		@Override
		public boolean defaultWeight() {
			return defaultWeight;
		}

		static int word(int idx) {
			return idx / WordSize;
		}

		static long bit(int idx) {
			return 1L << (idx % WordSize);
		}

		static int wordsNum(int size) {
			return (size + WordSize - 1) / WordSize;
		}

		void setImpl(int idx, boolean weight) {
			if (weight) {
				weights[word(idx)] |= bit(idx);
			} else {
				weights[word(idx)] &= ~bit(idx);
			}
		}
	}

	static class IndexImmutable  extends WeightsImplBool.IndexImpl  implements WeightsImpl.IndexImmutable<Boolean> {

		IndexImmutable(IWeightsBool  orig, IntSet elements, boolean isVertices) {
			super(orig, elements, isVertices);
		}

		IndexImmutable(IWeightsBool  orig, IntSet elements, boolean isVertices, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(orig, elements, isVertices, reIndexMap);
		}

		@Override
		public void set(int idx, boolean weight) {
			throw new UnsupportedOperationException("immutable weights");
		}
	}

	static class IndexMutable  extends WeightsImplBool.IndexImpl  implements WeightsImpl.IndexMutable<Boolean> {

		IndexMutable(IntSet elements, boolean isVertices, boolean defVal) {
			super(elements, isVertices, defVal);
		}

		IndexMutable(IWeightsBool  orig, IntSet elements, boolean isVertices) {
			super(orig, elements, isVertices);
		}

		@Override
		public void set(int idx, boolean weight) {
			checkIdx(idx);
			setImpl(idx, weight);
		}

		@Override
		public int capacity() {
			return weights.length * WordSize;
		}

		@Override
		public void expand(int newCapacity) {
			int oldCapacity = capacity();
			if (oldCapacity >= newCapacity)
				return;
			weights = Arrays.copyOf(weights, wordsNum(newCapacity));
			if (defaultWeight)
				Arrays.fill(weights, oldCapacity / WordSize, weights.length, ~0L);
		}

		@Override
		public void swapAndClear(int removedIdx, int swappedIdx) {
			// checkIdx(removedIdx);
			// checkIdx(swappedIdx);
			setImpl(removedIdx, (weights[word(swappedIdx)] & bit(swappedIdx)) != 0);
			setImpl(swappedIdx, defaultWeight);
		}

		@Override
		public void clear(int idx) {
			setImpl(idx, defaultWeight);
		}

		@Override
		public void clear() {
			Arrays.fill(weights, 0, wordsNum(size()), defaultWeight ? ~0L : 0);
		}
	}

	static class IntMapped  extends WeightsImpl.IntMapped<Boolean> implements IWeightsBool  {
		IntMapped(IWeightsBool  weights, IndexIntIdMap indexMap) {

			return (IWeightsBool ) super.weights();
		}

		@Override
		public boolean get(int id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(int id, boolean weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public boolean defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class IndexMasked  extends WeightsImpl.IndexAbstract<Boolean> implements IWeightsBool  {

		private final IWeightsBool  orig;
		private final Int2IntMap api2orig;

		IndexMasked(IWeightsBool  orig, IntSet elements, boolean isVertices, Int2IntMap api2orig) {
			super(elements, isVertices);
			this.orig = orig;
			this.api2orig = api2orig;
		}

		private int api2orig(int idx) {
			return api2orig.getOrDefault(idx, idx);
		}

		@Override
		public boolean get(int idx) {
			checkIdx(idx);
			return orig.get(api2orig(idx));
		}

		@Override
		public void set(int idx, boolean weight) {
			checkIdx(idx);
			orig.set(api2orig(idx), weight);
		}

		@Override
		public boolean defaultWeight() {
			return orig.defaultWeight();
		}
	}

	static class IntMappedMasked  implements IWeightsBool  {

		private final IWeightsBool  orig;
		private final IntSet mask;
		private final boolean isVertices;

		IntMappedMasked(IWeightsBool  orig, IntSet mask, boolean isVertices) {
			this.orig = orig;
			this.mask = mask;
			this.isVertices = isVertices;
		}

		private void checkId(int id) {
			if (mask.contains(id)) {
				if (isVertices) {
					throw NoSuchVertexException.ofVertex(id);
				} else {
					throw NoSuchEdgeException.ofEdge(id);
				}
			}
		}

		@Override
		public boolean get(int id) {
			checkId(id);
			return orig.get(id);
		}

		@Override
		public void set(int idx, boolean weight) {
			checkId(idx);
			orig.set(idx, weight);
		}

		@Override
		public boolean defaultWeight() {
			return orig.defaultWeight();
		}
	}

	static class ObjMappedMasked<K > implements WeightsBool<K > {

		private final WeightsBool<K > orig;
		private final ObjectSet<K> mask;
		private final boolean isVertices;

		ObjMappedMasked(WeightsBool<K > orig, ObjectSet<K> mask, boolean isVertices) {
			this.orig = orig;
			this.mask = mask;
			this.isVertices = isVertices;
		}

		private void checkId(K id) {
			if (mask.contains(id)) {
				if (isVertices) {
					throw NoSuchVertexException.ofVertex(id);
				} else {
					throw NoSuchEdgeException.ofEdge(id);
				}
			}
		}

		@Override
		public boolean get(K id) {
			checkId(id);
			return orig.get(id);
		}

		@Override
		public void set(K idx, boolean weight) {
			checkId(idx);
			orig.set(idx, weight);
		}

		@Override
		public boolean defaultWeight() {
			return orig.defaultWeight();
		}
	}
