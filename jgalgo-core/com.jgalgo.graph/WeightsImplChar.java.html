class WeightsImplChar {

	private WeightsImplChar() {}

	abstract static class IndexImpl  extends WeightsImpl.IndexAbstract<Character> implements IWeightsChar  {

		char[] weights;
		final char defaultWeight;


		IndexImpl(IntSet elements, boolean isVertices, char defVal) {
			super(elements, isVertices);
			defaultWeight = defVal;
			weights = CharArrays.EMPTY_ARRAY;
			if (!elements.isEmpty()) {
				weights = new char[elements.size()];
				Arrays.fill(weights, defaultWeight);
			}
		}

		IndexImpl(IWeightsChar  orig, IntSet elements, boolean isVertices) {
			super(elements, isVertices);
			defaultWeight = orig.defaultWeight();
			final int n = elements.size();

			if (orig instanceof WeightsImplChar.IndexImpl) {
				WeightsImplChar.IndexImpl  orig0 = (WeightsImplChar.IndexImpl ) orig;
				WeightsImpl.checkSameSize(elements, orig0.elements);
				weights = Arrays.copyOf(orig0.weights, n);

			} else {
				weights = CharArrays.EMPTY_ARRAY;
				if (n > 0) {
					weights = new char[n];
					for (int i : range(n))
						weights[i] = orig.get(i);
				}
			}
		}

		IndexImpl(IWeightsChar  orig, IntSet elements, boolean isVertices, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(elements, isVertices);
			defaultWeight = orig.defaultWeight();
			final int n = elements.size();
			weights = new char[n];

			if (orig instanceof WeightsImplChar.IndexImpl) {
				WeightsImplChar.IndexImpl  orig0 = (WeightsImplChar.IndexImpl ) orig;
				WeightsImpl.checkSameSize(elements, orig0.elements);
				for (int oldIdx : range(n))
					weights[reIndexMap.map(oldIdx)] = orig0.weights[oldIdx];
			} else {
				for (int oldIdx : range(n))
					weights[reIndexMap.map(oldIdx)] = orig.get(oldIdx);
			}
		}

		
		@Override
		public char get(int idx) {
			checkIdx(idx);
			return  weights[idx];
		}

		@Override
		public char defaultWeight() {
			return defaultWeight;
		}

	}

	static class IndexImmutable  extends WeightsImplChar.IndexImpl  implements WeightsImpl.IndexImmutable<Character> {

		IndexImmutable(IWeightsChar  orig, IntSet elements, boolean isVertices) {
			super(orig, elements, isVertices);
		}

		IndexImmutable(IWeightsChar  orig, IntSet elements, boolean isVertices, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(orig, elements, isVertices, reIndexMap);
		}

		@Override
		public void set(int idx, char weight) {
			throw new UnsupportedOperationException("immutable weights");
		}
	}

	static class IndexMutable  extends WeightsImplChar.IndexImpl  implements WeightsImpl.IndexMutable<Character> {

		IndexMutable(IntSet elements, boolean isVertices, char defVal) {
			super(elements, isVertices, defVal);
		}

		IndexMutable(IWeightsChar  orig, IntSet elements, boolean isVertices) {
			super(orig, elements, isVertices);
		}

		@Override
		public void set(int idx, char weight) {
			checkIdx(idx);
			weights[idx] = weight;
		}

		@Override
		public int capacity() {
			return weights.length;
		}

		@Override
		public void expand(int newCapacity) {
			int oldCapacity = capacity();
			assert oldCapacity < newCapacity;
			weights = Arrays.copyOf(weights, newCapacity);
			Arrays.fill(weights, oldCapacity, newCapacity, defaultWeight);
		}

		@Override
		public void swapAndClear(int removedIdx, int swappedIdx) {
			// checkIdx(removedIdx);
			// checkIdx(swappedIdx);
			weights[removedIdx] = weights[swappedIdx];
			weights[swappedIdx] = defaultWeight;
		}

		@Override
		public void clear(int idx) {
			weights[idx] = defaultWeight;
		}

		@Override
		public void clear() {
			Arrays.fill(weights, 0, size(), defaultWeight);
		}
	}

	static class IntMapped  extends WeightsImpl.IntMapped<Character> implements IWeightsChar  {
		IntMapped(IWeightsChar  weights, IndexIntIdMap indexMap) {

		}

		@Override
		public void set(int id, char weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public char defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class IndexMasked  extends WeightsImpl.IndexAbstract<Character> implements IWeightsChar  {

		private final IWeightsChar  orig;
		private final Int2IntMap api2orig;

		IndexMasked(IWeightsChar  orig, IntSet elements, boolean isVertices, Int2IntMap api2orig) {
			super(elements, isVertices);
			this.orig = orig;
			this.api2orig = api2orig;
		}

		private int api2orig(int idx) {
			return api2orig.getOrDefault(idx, idx);
		}

		@Override
		public char get(int idx) {
			checkIdx(idx);
			return orig.get(api2orig(idx));
		}

		@Override
		public void set(int idx, char weight) {
			checkIdx(idx);
			orig.set(api2orig(idx), weight);
		}

		@Override
		public char defaultWeight() {
			return orig.defaultWeight();
		}
	}

	static class IntMappedMasked  implements IWeightsChar  {

		private final IWeightsChar  orig;
		private final IntSet mask;
		private final boolean isVertices;

		IntMappedMasked(IWeightsChar  orig, IntSet mask, boolean isVertices) {
			this.orig = orig;
			this.mask = mask;
			this.isVertices = isVertices;
		}

		private void checkId(int id) {
			if (mask.contains(id)) {
				if (isVertices) {
					throw NoSuchVertexException.ofVertex(id);
				} else {
					throw NoSuchEdgeException.ofEdge(id);
				}
			}
		}

		@Override
		public char get(int id) {
			checkId(id);
			return orig.get(id);
		}

		@Override
		public void set(int idx, char weight) {
			checkId(idx);
			orig.set(idx, weight);
		}

		@Override
		public char defaultWeight() {
			return orig.defaultWeight();
		}
	}

		@Override

			checkId(idx);

		}

		@Override

			return orig.defaultWeight();
		}
	}
