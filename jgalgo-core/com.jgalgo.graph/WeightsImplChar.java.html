package com.jgalgo.graph;

import java.util.Arrays;
import it.unimi.dsi.fastutil.chars.CharArrays;

class WeightsImplChar {

	static abstract class IndexImpl  extends WeightsImpl.IndexAbstract<Character> implements WeightsChar  {

		char[] weights;
		final char defaultWeight;

		IndexImpl(GraphElementSet elements, char defVal) {
			super(elements);

			weights = CharArrays.EMPTY_ARRAY;
			defaultWeight = defVal;
		}

		IndexImpl(WeightsImplChar.IndexImpl  orig, GraphElementSet elements) {
			this(elements, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(orig.weights, elements.size());
		}

		IndexImpl(WeightsImplChar.IndexImpl  orig, GraphElementSet elements, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			this(elements, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(weights, elements.size());
			final int s = elements.size();
			for (int i = 0; i < s; i++)
				weights[reIndexMap.origToReIndexed(i)] = orig.weights[i];
		}

		
		@Override
		public char get(int idx) {
			checkIdx(idx);
			return  weights[idx];
		}

		@Override
		public char defaultWeight() {
			return defaultWeight;
		}
	}

	static class IndexImmutable  extends WeightsImplChar.IndexImpl  implements WeightsImpl.IndexImmutable<Character> {

		IndexImmutable(WeightsImplChar.IndexImpl  orig, GraphElementSet.FixedSize elements) {
			super(orig, elements);
		}

		IndexImmutable(WeightsImplChar.IndexImpl  orig, GraphElementSet.FixedSize elements,
				IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(orig, elements, reIndexMap);
		}

		@Override
		public void set(int idx, char weight) {
			throw new UnsupportedOperationException("immutable weights");
		}
	}

	static class IndexMutable  extends WeightsImplChar.IndexImpl  implements WeightsImpl.IndexMutable<Character> {

		IndexMutable(GraphElementSet elements, char defVal) {
			super(elements, defVal);
		}

		IndexMutable(WeightsImplChar.IndexImpl  orig, GraphElementSet elements) {
			super(orig, elements);
		}

		@Override
		public void set(int idx, char weight) {
			checkIdx(idx);
			weights[idx] = weight;
		}

		@Override
		public int capacity() {
			return weights.length;
		}

		@Override
		public void expand(int newCapacity) {
			int oldCapacity = weights.length;
			assert oldCapacity < newCapacity;
			weights = Arrays.copyOf(weights, newCapacity);
			Arrays.fill(weights, oldCapacity, newCapacity, defaultWeight);
		}

		@Override
		public void swap(int idx1, int idx2) {
			checkIdx(idx1);
			checkIdx(idx2);
			CharArrays.swap(weights, idx1, idx2);
		}

		@Override
		public void clear(int idx) {
			weights[idx] = defaultWeight;
		}

		@Override
		public void clear() {
			Arrays.fill(weights, 0, size(), defaultWeight);
		}
	}

	static class Mapped  extends WeightsImpl.Mapped<Character> implements WeightsChar  {
		Mapped(WeightsImplChar.IndexImpl  weights, IndexIdMap indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplChar.IndexImpl  weights() {
			return (WeightsImplChar.IndexImpl ) super.weights();
		}

		@Override
		public char get(int id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(int id, char weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public char defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ImmutableView  extends WeightsImpl.ImmutableView<Character> implements WeightsChar  {
		ImmutableView(WeightsChar  w) {
			super(w);
		}

		@Override
		WeightsChar  weights() {
			return (WeightsChar ) super.weights();
		}

		@Override
		public char get(int id) {
			return weights().get(id);
		}

		@Override
		public void set(int id, char weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public char defaultWeight() {
			return weights().defaultWeight();
		}
	}

}
