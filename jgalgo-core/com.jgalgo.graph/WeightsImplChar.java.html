package com.jgalgo.graph;

import static com.jgalgo.internal.util.Range.range;
import java.util.Arrays;
import it.unimi.dsi.fastutil.chars.CharArrays;
import it.unimi.dsi.fastutil.ints.IntSet;

class WeightsImplChar {

	private WeightsImplChar() {}

	abstract static class IndexImpl  extends WeightsImpl.IndexAbstract<Character> implements IWeightsChar  {

		char[] weights;
		final char defaultWeight;


		IndexImpl(IntSet elements, boolean isVertices, char defVal) {
			super(elements, isVertices);

			defaultWeight = defVal;

			weights = CharArrays.EMPTY_ARRAY;
			if (!elements.isEmpty()) {
				weights = Arrays.copyOf(weights, elements.size());
				Arrays.fill(weights, defaultWeight);
			}
		}

		IndexImpl(WeightsImplChar.IndexImpl  orig, IntSet elements, boolean isVertices) {
			this(elements, isVertices, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(orig.weights, elements.size());
		}

		IndexImpl(WeightsImplChar.IndexImpl  orig, IntSet elements, boolean isVertices, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			this(elements, isVertices, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(weights, elements.size());
			for (int i : range(elements.size()))
				weights[reIndexMap.map(i)] = orig.weights[i];
		}

		
		@Override
		public char get(int idx) {
			checkIdx(idx);
			return  weights[idx];
		}

		@Override
		public char defaultWeight() {
			return defaultWeight;
		}

	}

	static class IndexImmutable  extends WeightsImplChar.IndexImpl  implements WeightsImpl.IndexImmutable<Character> {

		IndexImmutable(WeightsImplChar.IndexImpl  orig, IntSet elements, boolean isVertices) {
			super(orig, elements, isVertices);
		}

		IndexImmutable(WeightsImplChar.IndexImpl  orig, IntSet elements, boolean isVertices, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(orig, elements, isVertices, reIndexMap);
		}

		@Override
		public void set(int idx, char weight) {
			throw new UnsupportedOperationException("immutable weights");
		}
	}

	static class IndexMutable  extends WeightsImplChar.IndexImpl  implements WeightsImpl.IndexMutable<Character> {

		IndexMutable(IntSet elements, boolean isVertices, char defVal) {
			super(elements, isVertices, defVal);
		}

		IndexMutable(WeightsImplChar.IndexImpl  orig, IntSet elements, boolean isVertices) {
			super(orig, elements, isVertices);
		}

		@Override
		public void set(int idx, char weight) {
			checkIdx(idx);
			weights[idx] = weight;
		}


		@Override
		public int capacity() {
			return weights.length;
		}

		@Override
		public void expand(int newCapacity) {
			int oldCapacity = capacity();
			assert oldCapacity < newCapacity;
			weights = Arrays.copyOf(weights, newCapacity);
			Arrays.fill(weights, oldCapacity, newCapacity, defaultWeight);
		}

		@Override
		public void swapAndClear(int removedIdx, int swappedIdx) {
			// checkIdx(removedIdx);
			// checkIdx(swappedIdx);
			weights[removedIdx] = weights[swappedIdx];
			weights[swappedIdx] = defaultWeight;
		}

		@Override
		public void clear(int idx) {
			weights[idx] = defaultWeight;
		}

		@Override
		public void clear() {
			Arrays.fill(weights, 0, size(), defaultWeight);
		}
	}

	static class IntMapped  extends WeightsImpl.IntMapped<Character> implements IWeightsChar  {
		IntMapped(WeightsImplChar.IndexImpl  weights, IndexIntIdMap indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplChar.IndexImpl  weights() {
			return (WeightsImplChar.IndexImpl ) super.weights();
		}

		@Override
		public char get(int id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(int id, char weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public char defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjMapped<K > extends WeightsImpl.ObjMapped<K, Character> implements WeightsChar<K > {
		ObjMapped(WeightsImplChar.IndexImpl  weights, IndexIdMap<K> indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplChar.IndexImpl  weights() {
			return (WeightsImplChar.IndexImpl ) super.weights();
		}

		@Override
		public char get(K id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(K id, char weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public char defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class IntImmutableView  extends WeightsImpl.IntImmutableView<Character> implements IWeightsChar  {
		IntImmutableView(IWeightsChar  w) {
			super(w);
		}

		@Override
		IWeightsChar  weights() {
			return (IWeightsChar ) super.weights();
		}

		@Override
		public char get(int id) {
			return weights().get(id);
		}

		@Override
		public void set(int id, char weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public char defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjImmutableView<K > extends WeightsImpl.ObjImmutableView<K, Character> implements WeightsChar <K > {
		ObjImmutableView(WeightsChar <K > w) {
			super(w);
		}

		@Override
		WeightsChar <K > weights() {
			return (WeightsChar <K >) super.weights();
		}

		@Override
		public char get(K id) {
			return weights().get(id);
		}

		@Override
		public void set(K id, char weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public char defaultWeight() {
			return weights().defaultWeight();
		}
	}

}
