package com.jgalgo.graph;

/**
 * Specific weights of {@code boolean} for {@linkplain IntGraph int graphs}.
 *
 * @author Barak Ugav
 */
public interface IWeightsBool  extends IWeights<Boolean>, WeightsBool<Integer >  {

	/**
	 * Get the weight associated with the given element.
	 *
	 * @param  element an element (edge/vertex.)
	 * @return    the weight associated with the given element.
	 */
	public boolean get(int element);

	@Deprecated
	@Override
	default boolean get(Integer element) {
		return get(element.intValue());
	}

	@Deprecated
	@Override
	default Boolean getAsObj(int element) {
		return Boolean.valueOf(get(element));
	}

	/**
	 * Set the weight associated with the given element.
	 *
	 * @param element     an element (edge/vertex)
	 * @param weight new weight that will be associated with the given element
	 */
	public void set(int element, boolean weight);

	@Deprecated
	@Override
	default void set(Integer element, boolean weight) {
		set(element.intValue(), weight);
	}

	@Deprecated
	@Override
	default void setAsObj(int element, Boolean weight) {
		set(element, weight.booleanValue());
	}

	@Deprecated
	@Override
	default Boolean getAsObj(Integer element) {
		return Boolean.valueOf(get(element.intValue()));
	}

	@Deprecated
	@Override
	default void setAsObj(Integer element, Boolean weight) {
		set(element.intValue(), weight.booleanValue());
	}

	

}
