package com.jgalgo.graph;

import java.util.Arrays;
import it.unimi.dsi.fastutil.objects.ObjectArrays;

class WeightsImplObj {

	abstract static class IndexImpl <T> extends WeightsImpl.IndexAbstract<T> implements IWeightsObj <T> {

		Object[] weights;
		final T defaultWeight;

		IndexImpl(GraphElementSet elements, T defVal) {
			super(elements);

			weights = ObjectArrays.EMPTY_ARRAY;
			defaultWeight = defVal;
		}

		IndexImpl(WeightsImplObj.IndexImpl <T> orig, GraphElementSet elements) {
			this(elements, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(orig.weights, elements.size());
		}

		IndexImpl(WeightsImplObj.IndexImpl <T> orig, GraphElementSet elements,
				IndexGraphBuilder.ReIndexingMap reIndexMap) {
			this(elements, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(weights, elements.size());
			final int s = elements.size();
			for (int i = 0; i < s; i++)
				weights[reIndexMap.origToReIndexed(i)] = orig.weights[i];
		}

		@SuppressWarnings("unchecked")
		@Override
		public T get(int idx) {
			checkIdx(idx);
			return (T) weights[idx];
		}

		@Override
		public T defaultWeight() {
			return defaultWeight;
		}
	}

	static class IndexImmutable <T> extends WeightsImplObj.IndexImpl <T> implements WeightsImpl.IndexImmutable<T> {

		IndexImmutable(WeightsImplObj.IndexImpl <T> orig, GraphElementSet.FixedSize elements) {
			super(orig, elements);
		}

		IndexImmutable(WeightsImplObj.IndexImpl <T> orig, GraphElementSet.FixedSize elements,
				IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(orig, elements, reIndexMap);
		}

		@Override
		public void set(int idx, T weight) {
			throw new UnsupportedOperationException("immutable weights");
		}
	}

	static class IndexMutable <T> extends WeightsImplObj.IndexImpl <T> implements WeightsImpl.IndexMutable<T> {

		IndexMutable(GraphElementSet elements, T defVal) {
			super(elements, defVal);
		}

		IndexMutable(WeightsImplObj.IndexImpl <T> orig, GraphElementSet elements) {
			super(orig, elements);
		}

		@Override
		public void set(int idx, T weight) {
			checkIdx(idx);
			weights[idx] = weight;
		}

		@Override
		public int capacity() {
			return weights.length;
		}

		@Override
		public void expand(int newCapacity) {
			int oldCapacity = weights.length;
			assert oldCapacity < newCapacity;
			weights = Arrays.copyOf(weights, newCapacity);
			Arrays.fill(weights, oldCapacity, newCapacity, defaultWeight);
		}

		@Override
		public void swap(int idx1, int idx2) {
			checkIdx(idx1);
			checkIdx(idx2);
			ObjectArrays.swap(weights, idx1, idx2);
		}

		@Override
		public void clear(int idx) {
			weights[idx] = defaultWeight;
		}

		@Override
		public void clear() {
			Arrays.fill(weights, 0, size(), defaultWeight);
		}
	}

	static class IntMapped <T> extends WeightsImpl.IntMapped<T> implements IWeightsObj <T> {
		IntMapped(WeightsImplObj.IndexImpl <T> weights, IndexIntIdMap indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplObj.IndexImpl <T> weights() {
			return (WeightsImplObj.IndexImpl <T>) super.weights();
		}

		@Override
		public T get(int id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(int id, T weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public T defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjMapped<K , T> extends WeightsImpl.ObjMapped<K, T> implements WeightsObj<K , T> {
		ObjMapped(WeightsImplObj.IndexImpl <T> weights, IndexIdMap<K> indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplObj.IndexImpl <T> weights() {
			return (WeightsImplObj.IndexImpl <T>) super.weights();
		}

		@Override
		public T get(K id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(K id, T weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public T defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class IntImmutableView <T> extends WeightsImpl.IntImmutableView<T> implements IWeightsObj <T> {
		IntImmutableView(IWeightsObj <T> w) {
			super(w);
		}

		@Override
		IWeightsObj <T> weights() {
			return (IWeightsObj <T>) super.weights();
		}

		@Override
		public T get(int id) {
			return weights().get(id);
		}

		@Override
		public void set(int id, T weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public T defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjImmutableView<K , T> extends WeightsImpl.ObjImmutableView<K, T> implements WeightsObj <K , T> {
		ObjImmutableView(WeightsObj <K , T> w) {
			super(w);
		}

		@Override
		WeightsObj <K , T> weights() {
			return (WeightsObj <K , T>) super.weights();
		}

		@Override
		public T get(K id) {
			return weights().get(id);
		}

		@Override
		public void set(K id, T weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public T defaultWeight() {
			return weights().defaultWeight();
		}
	}

}
