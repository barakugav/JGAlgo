package com.jgalgo.graph;

/**
 * Specific weights of {@code float} for {@linkplain IntGraph int graphs}.
 *
 * @author Barak Ugav
 */
public interface IWeightsFloat  extends IWeights<Float>, WeightsFloat<Integer >
, IWeightFunction
{

	/**
	 * Get the weight associated with the given element.
	 *
	 * @param  element an element (edge/vertex.)
	 * @return    the weight associated with the given element.
	 */
	public float get(int element);

	/**
	 * {@inheritDoc}
	 *
	 * @deprecated Please use {@link #get(int)} instead to avoid un/boxing.
	 */
	@Deprecated
	@Override
	default float get(Integer element) {
		return get(element.intValue());
	}

	/**
	 * {@inheritDoc}
	 *
	 * @deprecated Please use {@link #get(int)} instead to avoid un/boxing.
	 */
	@Deprecated
	@Override
	default Float getAsObj(int element) {
		return Float.valueOf(get(element));
	}

	/**
	 * {@inheritDoc}
	 *
	 * @deprecated Please use {@link #get(int)} instead to avoid un/boxing.
	 */
	@Deprecated
	@Override
	default Float getAsObj(Integer element) {
		return Float.valueOf(get(element.intValue()));
	}

	/**
	 * Set the weight associated with the given element.
	 *
	 * @param element     an element (edge/vertex)
	 * @param weight new weight that will be associated with the given element
	 */
	public void set(int element, float weight);

	/**
	 * {@inheritDoc}
	 *
	 * @deprecated Please use {@link #set(int, float)} instead to avoid un/boxing.
	 */
	@Deprecated
	@Override
	default void set(Integer element, float weight) {
		set(element.intValue(), weight);
	}

	/**
	 * {@inheritDoc}
	 *
	 * @deprecated Please use {@link #set(int, float)} instead to avoid un/boxing.
	 */
	@Deprecated
	@Override
	default void setAsObj(int element, Float weight) {
		set(element, weight.floatValue());
	}

	/**
	 * {@inheritDoc}
	 *
	 * @deprecated Please use {@link #set(int, float)} instead to avoid un/boxing.
	 */
	@Deprecated
	@Override
	default void setAsObj(Integer element, Float weight) {
		set(element.intValue(), weight.floatValue());
	}

	/**
	 * {@inheritDoc}
	 *
	 * <p>
	 * Implement the {@link IWeightFunction} interface by using the weights of the container.
	 */
	@Override
	default double weight(int element) {
		return get(element);
	}

	/**
	 * {@inheritDoc}
	 *
	 * <p>
	 * Implement the {@link WeightFunctionInt} interface by using the weights of the container.
	 */
	@Deprecated
	@Override
	default double weight(Integer element) {
		return get(element.intValue());
	}

	@Deprecated
	@Override
	default int compare(Integer e1, Integer e2) {
		return IWeightFunction.super.compare(e1.intValue(), e2.intValue());
	}

}
