package com.jgalgo.graph;

import java.util.Arrays;
import it.unimi.dsi.fastutil.floats.FloatArrays;

class WeightsImplFloat {

	static abstract class IndexImpl  extends WeightsImpl.IndexAbstract<Float> implements IWeightsFloat  {

		float[] weights;
		final float defaultWeight;

		IndexImpl(GraphElementSet elements, float defVal) {
			super(elements);

			weights = FloatArrays.EMPTY_ARRAY;
			defaultWeight = defVal;
		}

		IndexImpl(WeightsImplFloat.IndexImpl  orig, GraphElementSet elements) {
			this(elements, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(orig.weights, elements.size());
		}

		IndexImpl(WeightsImplFloat.IndexImpl  orig, GraphElementSet elements, IndexGraphBuilder.ReIndexingMap reIndexMap) {
			this(elements, orig.defaultWeight);
			WeightsImpl.checkSameSize(elements, orig.elements);
			weights = Arrays.copyOf(weights, elements.size());
			final int s = elements.size();
			for (int i = 0; i < s; i++)
				weights[reIndexMap.origToReIndexed(i)] = orig.weights[i];
		}

		
		@Override
		public float get(int idx) {
			checkIdx(idx);
			return  weights[idx];
		}

		@Override
		public float defaultWeight() {
			return defaultWeight;
		}
	}

	static class IndexImmutable  extends WeightsImplFloat.IndexImpl  implements WeightsImpl.IndexImmutable<Float> {

		IndexImmutable(WeightsImplFloat.IndexImpl  orig, GraphElementSet.FixedSize elements) {
			super(orig, elements);
		}

		IndexImmutable(WeightsImplFloat.IndexImpl  orig, GraphElementSet.FixedSize elements,
				IndexGraphBuilder.ReIndexingMap reIndexMap) {
			super(orig, elements, reIndexMap);
		}

		@Override
		public void set(int idx, float weight) {
			throw new UnsupportedOperationException("immutable weights");
		}
	}

	static class IndexMutable  extends WeightsImplFloat.IndexImpl  implements WeightsImpl.IndexMutable<Float> {

		IndexMutable(GraphElementSet elements, float defVal) {
			super(elements, defVal);
		}

		IndexMutable(WeightsImplFloat.IndexImpl  orig, GraphElementSet elements) {
			super(orig, elements);
		}

		@Override
		public void set(int idx, float weight) {
			checkIdx(idx);
			weights[idx] = weight;
		}

		@Override
		public int capacity() {
			return weights.length;
		}

		@Override
		public void expand(int newCapacity) {
			int oldCapacity = weights.length;
			assert oldCapacity < newCapacity;
			weights = Arrays.copyOf(weights, newCapacity);
			Arrays.fill(weights, oldCapacity, newCapacity, defaultWeight);
		}

		@Override
		public void swap(int idx1, int idx2) {
			checkIdx(idx1);
			checkIdx(idx2);
			FloatArrays.swap(weights, idx1, idx2);
		}

		@Override
		public void clear(int idx) {
			weights[idx] = defaultWeight;
		}

		@Override
		public void clear() {
			Arrays.fill(weights, 0, size(), defaultWeight);
		}
	}

	static class IntMapped  extends WeightsImpl.IntMapped<Float> implements IWeightsFloat  {
		IntMapped(WeightsImplFloat.IndexImpl  weights, IndexIntIdMap indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplFloat.IndexImpl  weights() {
			return (WeightsImplFloat.IndexImpl ) super.weights();
		}

		@Override
		public float get(int id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(int id, float weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public float defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjMapped<K > extends WeightsImpl.ObjMapped<K, Float> implements WeightsFloat<K > {
		ObjMapped(WeightsImplFloat.IndexImpl  weights, IndexIdMap<K> indexMap) {
			super(weights, indexMap);
		}

		@Override
		public WeightsImplFloat.IndexImpl  weights() {
			return (WeightsImplFloat.IndexImpl ) super.weights();
		}

		@Override
		public float get(K id) {
			return weights().get(indexMap.idToIndex(id));
		}

		@Override
		public void set(K id, float weight) {
			weights().set(indexMap.idToIndex(id), weight);
		}

		@Override
		public float defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class IntImmutableView  extends WeightsImpl.IntImmutableView<Float> implements IWeightsFloat  {
		IntImmutableView(IWeightsFloat  w) {
			super(w);
		}

		@Override
		IWeightsFloat  weights() {
			return (IWeightsFloat ) super.weights();
		}

		@Override
		public float get(int id) {
			return weights().get(id);
		}

		@Override
		public void set(int id, float weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public float defaultWeight() {
			return weights().defaultWeight();
		}
	}

	static class ObjImmutableView<K > extends WeightsImpl.ObjImmutableView<K, Float> implements WeightsFloat <K > {
		ObjImmutableView(WeightsFloat <K > w) {
			super(w);
		}

		@Override
		WeightsFloat <K > weights() {
			return (WeightsFloat <K >) super.weights();
		}

		@Override
		public float get(K id) {
			return weights().get(id);
		}

		@Override
		public void set(K id, float weight) {
			throw new UnsupportedOperationException("immutable weights");
		}

		@Override
		public float defaultWeight() {
			return weights().defaultWeight();
		}
	}

}
